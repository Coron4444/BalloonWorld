160 HRESULT InitSound(HWND hWnd)
161 {
162  HRESULT hr;
163 
164  // COMライブラリの初期化
165  CoInitializeEx(NULL, COINIT_MULTITHREADED);
166 
167  // XAudio2オブジェクトの作成
168  hr = XAudio2Create(&g_pXAudio2, 0);
169  if(FAILED(hr))
170  {
171  MessageBox(hWnd, "XAudio2オブジェクトの作成に失敗！", "警告！", MB_ICONWARNING);
172 
173  // COMライブラリの終了処理
174  CoUninitialize();
175 
176  return E_FAIL;
177  }
178 
179  // マスターボイスの生成
180  hr = g_pXAudio2->CreateMasteringVoice(&g_pMasteringVoice);
181  if(FAILED(hr))
182  {
183  MessageBox(hWnd, "マスターボイスの生成に失敗！", "警告！", MB_ICONWARNING);
184 
185  if(g_pXAudio2)
186  {
187  // XAudio2オブジェクトの開放
188  g_pXAudio2->Release();
189  g_pXAudio2 = NULL;
190  }
191 
192  // COMライブラリの終了処理
193  CoUninitialize();
194 
195  return E_FAIL;
196  }
197 
198  // サウンドデータの初期化
199  for(int nCntSound = 0; nCntSound < SOUND_LABEL_MAX; nCntSound++)
200  {
201  HANDLE hFile;
202  DWORD dwChunkSize = 0;
203  DWORD dwChunkPosition = 0;
204  DWORD dwFiletype;
205  WAVEFORMATEXTENSIBLE wfx;
206  XAUDIO2_BUFFER buffer;
207 
208  // バッファのクリア
209  memset(&wfx, 0, sizeof(WAVEFORMATEXTENSIBLE));
210  memset(&buffer, 0, sizeof(XAUDIO2_BUFFER));
211 
212  // サウンドデータファイルの生成
213  hFile = CreateFile(g_aParam[nCntSound].pFilename, GENERIC_READ, FILE_SHARE_READ, NULL, OPEN_EXISTING, 0, NULL);
214  if(hFile == INVALID_HANDLE_VALUE)
215  {
216  MessageBox(hWnd, "サウンドデータファイルの生成に失敗！(1)", "警告！", MB_ICONWARNING);
217  return HRESULT_FROM_WIN32(GetLastError());
218  }
219  if(SetFilePointer(hFile, 0, NULL, FILE_BEGIN) == INVALID_SET_FILE_POINTER)
220  {// ファイルポインタを先頭に移動
221  MessageBox(hWnd, "サウンドデータファイルの生成に失敗！(2)", "警告！", MB_ICONWARNING);
222  return HRESULT_FROM_WIN32(GetLastError());
223  }
224 
225  // WAVEファイルのチェック
226  hr = CheckChunk(hFile, 'FFIR', &dwChunkSize, &dwChunkPosition);
227  if(FAILED(hr))
228  {
229  MessageBox(hWnd, "WAVEファイルのチェックに失敗！(1)", "警告！", MB_ICONWARNING);
230  return S_FALSE;
231  }
232  hr = ReadChunkData(hFile, &dwFiletype, sizeof(DWORD), dwChunkPosition);
233  if(FAILED(hr))
234  {
235  MessageBox(hWnd, "WAVEファイルのチェックに失敗！(2)", "警告！", MB_ICONWARNING);
236  return S_FALSE;
237  }
238  if(dwFiletype != 'EVAW')
239  {
240  MessageBox(hWnd, "WAVEファイルのチェックに失敗！(3)", "警告！", MB_ICONWARNING);
241  return S_FALSE;
242  }
243 
244  // フォーマットチェック
245  hr = CheckChunk(hFile, ' tmf', &dwChunkSize, &dwChunkPosition);
246  if(FAILED(hr))
247  {
248  MessageBox(hWnd, "フォーマットチェックに失敗！(1)", "警告！", MB_ICONWARNING);
249  return S_FALSE;
250  }
251  hr = ReadChunkData(hFile, &wfx, dwChunkSize, dwChunkPosition);
252  if(FAILED(hr))
253  {
254  MessageBox(hWnd, "フォーマットチェックに失敗！(2)", "警告！", MB_ICONWARNING);
255  return S_FALSE;
256  }
257 
258  // オーディオデータ読み込み
259  hr = CheckChunk(hFile, 'atad', &g_aSizeAudio[nCntSound], &dwChunkPosition);
260  if(FAILED(hr))
261  {
262  MessageBox(hWnd, "オーディオデータ読み込みに失敗！(1)", "警告！", MB_ICONWARNING);
263  return S_FALSE;
264  }
265  g_apDataAudio[nCntSound] = (BYTE*)malloc(g_aSizeAudio[nCntSound]);
266  hr = ReadChunkData(hFile, g_apDataAudio[nCntSound], g_aSizeAudio[nCntSound], dwChunkPosition);
267  if(FAILED(hr))
268  {
269  MessageBox(hWnd, "オーディオデータ読み込みに失敗！(2)", "警告！", MB_ICONWARNING);
270  return S_FALSE;
271  }
272 
273  // ソースボイスの生成
274  hr = g_pXAudio2->CreateSourceVoice(&g_apSourceVoice[nCntSound], &(wfx.Format));
275  if(FAILED(hr))
276  {
277  MessageBox(hWnd, "ソースボイスの生成に失敗！", "警告！", MB_ICONWARNING);
278  return S_FALSE;
279  }
280 
281  // バッファの値設定
282  memset(&buffer, 0, sizeof(XAUDIO2_BUFFER));
283  buffer.AudioBytes = g_aSizeAudio[nCntSound];
284  buffer.pAudioData = g_apDataAudio[nCntSound];
285  buffer.Flags = XAUDIO2_END_OF_STREAM;
286  buffer.LoopCount = g_aParam[nCntSound].nCntLoop;
287 
288  // オーディオバッファの登録
289  g_apSourceVoice[nCntSound]->SubmitSourceBuffer(&buffer);
290  }
291 
292  return S_OK;
293 }

371 HRESULT PlaySound(SOUND_LABEL label)
372 {
373  XAUDIO2_VOICE_STATE xa2state;
374  XAUDIO2_BUFFER buffer;
375 
376  // バッファの値設定
377  memset(&buffer, 0, sizeof(XAUDIO2_BUFFER));
378  buffer.AudioBytes = g_aSizeAudio[label];
379  buffer.pAudioData = g_apDataAudio[label];
380  buffer.Flags = XAUDIO2_END_OF_STREAM;
381  buffer.LoopCount = g_aParam[label].nCntLoop;
382 
383  // 状態取得
384  g_apSourceVoice[label]->GetState(&xa2state);
385  if(xa2state.BuffersQueued != 0)
386  {// 再生中
387  // 一時停止
388  g_apSourceVoice[label]->Stop(0);
389 
390  // オーディオバッファの削除
391  g_apSourceVoice[label]->FlushSourceBuffers();
392  }
393 
394  // オーディオバッファの登録
395  g_apSourceVoice[label]->SubmitSourceBuffer(&buffer);
396 
397  // 再生
398  g_apSourceVoice[label]->Start(0);
399 
400  return S_OK;
401 }

495 HRESULT CheckChunk(HANDLE hFile, DWORD format, DWORD *pChunkSize, DWORD *pChunkDataPosition)
496 {
497  HRESULT hr = S_OK;
498  DWORD dwRead;
499  DWORD dwChunkType;
500  DWORD dwChunkDataSize;
501  DWORD dwRIFFDataSize = 0;
502  DWORD dwFileType;
503  DWORD dwBytesRead = 0;
504  DWORD dwOffset = 0;
505 
506  if(SetFilePointer(hFile, 0, NULL, FILE_BEGIN) == INVALID_SET_FILE_POINTER)
507  {// ファイルポインタを先頭に移動
508  return HRESULT_FROM_WIN32(GetLastError());
509  }
510 
511  while(hr == S_OK)
512  {
513  if(ReadFile(hFile, &dwChunkType, sizeof(DWORD), &dwRead, NULL) == 0)
514  {// チャンクの読み込み
515  hr = HRESULT_FROM_WIN32(GetLastError());
516  }
517 
518  if(ReadFile(hFile, &dwChunkDataSize, sizeof(DWORD), &dwRead, NULL) == 0)
519  {// チャンクデータの読み込み
520  hr = HRESULT_FROM_WIN32(GetLastError());
521  }
522 
523  switch(dwChunkType)
524  {
525  case 'FFIR':
526  dwRIFFDataSize = dwChunkDataSize;
527  dwChunkDataSize = 4;
528  if(ReadFile(hFile, &dwFileType, sizeof(DWORD), &dwRead, NULL) == 0)
529  {// ファイルタイプの読み込み
530  hr = HRESULT_FROM_WIN32(GetLastError());
531  }
532  break;
533 
534  default:
535  if(SetFilePointer(hFile, dwChunkDataSize, NULL, FILE_CURRENT) == INVALID_SET_FILE_POINTER)
536  {// ファイルポインタをチャンクデータ分移動
537  return HRESULT_FROM_WIN32(GetLastError());
538  }
539  }
540 
541  dwOffset += sizeof(DWORD) * 2;
542  if(dwChunkType == format)
543  {
544  *pChunkSize = dwChunkDataSize;
545  *pChunkDataPosition = dwOffset;
546 
547  return S_OK;
548  }
549 
550  dwOffset += dwChunkDataSize;
551  if(dwBytesRead >= dwRIFFDataSize)
552  {
553  return S_FALSE;
554  }
555  }
556 
557  return S_OK;
558 }
