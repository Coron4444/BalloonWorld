174  void SeparateInto3BitIntervals(DWORD* num)
175  {
176  *num = (*num | *num << 8) & 0x0000f00f; // 1Byteを4bitごとに分ける

187  DWORD GetMortonNumber(const Vec3* position)
188  {
189  // 最下位レベル空間での座標へ変換
190  DWORD temp_x = (DWORD)((position->x - octree_width_min_.x) / octree_unit_length_.x);
191  DWORD temp_y = (DWORD)((position->y - octree_width_min_.y) / octree_unit_length_.y);
192  DWORD temp_z = (DWORD)((position->z - octree_width_min_.z) / octree_unit_length_.z);
193 
194  // 3bit間隔へ分離
195  SeparateInto3BitIntervals(&temp_x);
196  SeparateInto3BitIntervals(&temp_y);
197  SeparateInto3BitIntervals(&temp_z);
198 
199  // 1bitずつずらす
200  return temp_x | (temp_y << 1) | (temp_z << 2);
201  }

210  DWORD GetArrayIndexOfBelonginSpace(const Vec3* object_min, const Vec3* object_max)
211  {
212  // 最小値と最大値のモートン番号のXORを算出
213  DWORD converted_min = GetMortonNumber(object_min);
214  DWORD converted_max = GetMortonNumber(object_max);
215  DWORD temp = converted_min ^ converted_max;
216 
217  // 最上位の区切りから最小値と最大値の共有空間レベルを算出する
218  unsigned shared_space_level = 0;
219  for (unsigned i = 0; i < lowest_level_; i++)
220  {
221  // 3bitずつずらして(3Dな為)何かしら数値があるかをチェック
222  DWORD check = temp >> (i * 3) & 0x7;
223  if (check != 0) shared_space_level = i + 1;
224  }
225 
226  // 共有空間レベルから所属空間番号を算出(共有空間レベル分3bitシフト)
227  DWORD shared_space_num = converted_max >> (shared_space_level * 3);
228 
229  // 所属空間番号から配列インデックスに変換(等比級数の和の公式の分を加算)
230  shared_space_num += (power_of_eight_array_[lowest_level_ - shared_space_level] - 1) / 7;
231 
232  return shared_space_num;
233  }

279

281  for (auto it = collision_stack->begin(); it != collision_stack->end(); it++)
282  {
283  collision_list->push_back(temp01->getObject());
284  collision_list->push_back(*it);
285  }
286 
287  // 次のオブジェクトへ
288  temp01 = temp01->getNextPointer();
289  }
290 
291  // 子空間へ移動
292  bool child_flag = false; // 子空間フラグ
293  DWORD object_num = 0;
294  DWORD next_space_index = 0;
295  for (unsigned i = 0; i < 8; i++)
296  {
297  // 次の空間があるかどうか
298  next_space_index = space_index * 8 + 1 + i;
299  if (next_space_index >= all_space_num_) continue;
300  if (space_pointer_array_[next_space_index] == nullptr) continue;
301 
302  // 親をプッシュ
303  if (!child_flag)
304  {
305  temp01 = space_pointer_array_[space_index]->getFirstObject();
306  while (temp01 != nullptr)
307  {
308  collision_stack->push_back(temp01->getObject());
309  object_num++;
310  temp01 = temp01->getNextPointer();
311  }
312  }
313 
314  // 子空間フラグON
315  child_flag = true;
316 
317  // 子空間の再帰処理
318  GetCollisionList(next_space_index, collision_list, collision_stack);
319  }
320 
321  // スタックからオブジェクトを外す
322  if (!child_flag) return;
323 
324  for (unsigned i = 0; i < object_num; i++)
325  {
326  collision_stack->pop_back();
327  }
