100 void EffectFile::InputEffectFile()
101 {
102  // デバイスの取得
103  LPDIRECT3DDEVICE9 device = RendererDirectX9::GetDevice();
104  assert(device != nullptr && "Deviceの取得に失敗(EffectFile.cpp)");
105 
106  // シェーダーの読み込み
107  FILE* fp;
108  fp = fopen("resource/HLSL/AnimationFBX.obj", "rb");
109 
110  if (fp == nullptr) return;
111 
112  // ファイルサイズ取得
113  fseek(fp, 0, SEEK_END);
114  unsigned size = (unsigned)ftell(fp);
115  fseek(fp, 0, SEEK_SET);
116 
117  // ファイルデータ取得
118  char* src_data = new char[size];
119  fread(src_data, size, sizeof(char), fp);
120  fclose(fp);
121 
122  LPD3DXBUFFER error = nullptr;
123  // エフェクトファイルのコンパイル
124  HRESULT hr = D3DXCreateEffect(device, // デバイス
125  src_data, // コンパイル済みfxデータ
126  size, // コンパイル済みfxデータのサイズ
127  NULL, // プリプロセッサ定義へのポインタ
128  NULL, // オプションのインタフェースへのポインタ
129  0, // D3DXSHADERで識別されるコンパイルオプション
130  NULL, // 共有引数で使うID3DXEffectPoolオブジェクトへのポインタ
131  &effect_, // コンパイルされたエフェクトの格納先
132  &error); // コンパイルエラー一覧の格納先
133 
134  if (FAILED(hr))
135  {
136  MessageBox( NULL, (LPCTSTR)error->GetBufferPointer(), "エラー", MB_OK );
137  return;
138  }
139 
140  // 各種開放
141  SafeRelease(&error);
142  delete[] src_data;
143 }
