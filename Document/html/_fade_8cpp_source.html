152 void Fade::Start(Type type, State state, Vec2 size, XColor4 color, float speed)

167  transform_->CreateWorldMatrix();
168 
169  // エンドフラグOFF
170  end_flag_ = false;
171 
172  // ステートごとの処理
173  color_ = color;
174  double distance = 0.0f;
175  if (state_ == State::FADE_IN)
176  {

179  plane_polygon_->setColor(color_);

184 
185  alpha = 0.0f;

187  inter_container_.push_back(alpha, distance);
188 
189  inter_iter_ = inter_container_.begin();
190  }
191  else if (state_ == State::FADE_OUT)
192  {
193  float alpha = 0.0f;
194  color_.a = alpha;
195  plane_polygon_->setColor(color_);
196

204
