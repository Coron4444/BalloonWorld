89 void StencilShadowTestDraw::SettingBeforeDrawing(Camera* camera, unsigned object_index)
90 {
91  LPDIRECT3DDEVICE9 device;
92  Renderer::getpInstance()->getDevice(&device);
93  switch (object_index)
94  {
95  case 0:
96  {
97  device->SetRenderState(D3DRS_ZWRITEENABLE, FALSE);
98  device->SetRenderState(D3DRS_STENCILENABLE, TRUE);
99  device->SetRenderState(D3DRS_COLORWRITEENABLE, 0);
100  device->SetRenderState(D3DRS_STENCILFUNC, D3DCMP_ALWAYS);
101  device->SetRenderState(D3DRS_STENCILPASS, D3DSTENCILOP_INCR);
102  device->SetRenderState(D3DRS_STENCILREF, 1);
103  device->SetRenderState(D3DRS_CULLMODE, D3DCULL_CCW);
104  break;
105  }
106  case 1:
107  {
108  device->SetRenderState(D3DRS_STENCILPASS, D3DSTENCILOP_DECR);
109  device->SetRenderState(D3DRS_CULLMODE, D3DCULL_CW);
110  break;
111  }
112  case 2:
113  {
114  DrawBase::LimitedChangeCameraType(camera, Camera::Type::TWO_DIMENSIONAL);
115  device->SetRenderState(D3DRS_CULLMODE, D3DCULL_CCW);
116  device->SetRenderState(D3DRS_COLORWRITEENABLE, 0xf);
117  device->SetRenderState(D3DRS_STENCILFUNC, D3DCMP_EQUAL);
118  break;
119  }
120  }
121 }
