298 inline int32_t ConvertUtf16ToUtf8( int8_t* dst, int32_t dst_size, const int16_t* src )
299 {
300  int32_t cnt = 0;
301  const int16_t* wp = src;
302  int8_t* cp = dst;
303 
304  if (dst_size == 0) return 0;
305 
306  dst_size -= 3;
307 
308  for (cnt = 0; cnt < dst_size; )
309  {
310  int16_t wc = *wp++;
311  if (wc == 0)
312  {
313  break;
314  }
315  if ((wc & ~0x7f) == 0)
316  {
317  *cp++ = wc & 0x7f;
318  cnt += 1;
319  } else if ((wc & ~0x7ff) == 0)
320  {
321  *cp++ = ((wc >> 6) & 0x1f) | 0xc0;
322  *cp++ = ((wc) & 0x3f) | 0x80;
323  cnt += 2;
324  } else {
325  *cp++ = ((wc >> 12) & 0xf) | 0xe0;
326  *cp++ = ((wc >> 6) & 0x3f) | 0x80;
327  *cp++ = ((wc) & 0x3f) | 0x80;

342 inline int32_t ConvertUtf8ToUtf16( int16_t* dst, int32_t dst_size, const int8_t* src )
343 {
344  int32_t i, code;
345  int8_t c0, c1, c2;
346 
347  if (dst_size == 0) return 0;
348 
349  dst_size -= 1;
350 
351  for (i = 0; i < dst_size; i++)
352  {
353  int16_t wc;
354 
355  c0 = *src++;
356  if (c0 == '\0')
357  {
358  break;
359  }
360  // UTF8からUTF16に変換
361  code = (uint8_t)c0 >> 4;
362  if (code <= 7)
363  {
364  // 8bit文字
365  wc = c0;
366  }
367  else if (code >= 12 && code <= 13)
368  {
369  // 16bit文字
370  c1 = *src++;
371  wc = ((c0 & 0x1F) << 6) | (c1 & 0x3F);
372  }
373  else if (code == 14)
374  {
375  // 24bit文字
376  c1 = *src++;
377  c2 = *src++;

382  continue;
383  }
384  dst[i] = wc;
