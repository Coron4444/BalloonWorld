140 void GameScene::Update()
141 {
142  UpdateState();
143 
144  // ポーズ
145  if (getCurrentStateID() == (int)State::END) return;
146  if (pause_->getIsDecision())
147  {
148  switch (pause_->getSelectType())
149  {
150  case Pause::SelectType::CONTINUE:
151  {
152  setIsPause(false);
153  pause_->setIsEnable(false);
154  break;
155  }
156  case Pause::SelectType::RESTART:
157  {
158  getpSceneManager()->setNextScene(new GameScene(new GameSceneState_Start()));
159  pause_->setIsEnable(false);
160  break;
161  }
162  case Pause::SelectType::QUIT:
163  {
164  getpSceneManager()->setNextScene(new TitleScene(new TitleSceneState_Start()));
165  pause_->setIsEnable(false);
166  break;
167  }
168  }
169  }
170  else
171  {
172  if (!pause_->getIsInput()) return;
173  if (InputManager::getpInstance()->getpKeyboard()->getTrigger(DIK_P) ||
174  InputManager::getpInstance()->getpController()
175  ->getRelease(0, XINPUT_GAMEPAD_START))
176  {
177  if (getIsPause())
178  {
179  setIsPause(false);
180  pause_->setIsEnable(false);
181  SoundManager::getpInstance()->PlayOrStop(SoundManager::Type::SE_OPEN);
182  }
183  else
184  {
185  setIsPause(true);
186  pause_->setIsEnable(true);
187  SoundManager::getpInstance()->PlayOrStop(SoundManager::Type::SE_CLOSE);
188  }
189  }
190  }
191 }
