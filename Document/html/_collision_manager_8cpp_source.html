293 void CollisionManager::CollisionGround()
294 {

296 
297  for (unsigned i = 0; i < all_collision_.getEndIndex(); i++)
298  {
299  // 地面との判定をするかどうか

301

304  ->getpGameObject()->getpTransform()
305  ->getpPosition());
306 
307  // 衝突してるか
308  if (position_y >= all_collision_.getObject(i)
309  ->getpGameObject()->getpTransform()->getpPosition()->y)
310  {
311  // 地面との衝突処理
312  all_collision_.getObject(i)->HitGround(position_y);

319 void CollisionManager::ActualCalculation(CollisionObject* collision_object0,
320  CollisionObject* collision_object1)
321 {
322  for (unsigned i = 0; i < collision_object0->getEndIndexOfArray(); i++)
323  {
324  // 総当たり
325  for (unsigned j = 0; j < collision_object1->getEndIndexOfArray(); j++)
326  {
327  // 衝突計算
328  CollisionInformation information0;
329  CollisionInformation information1;
330  if (SortCollisionCalculation(&information0, &information1,
331  collision_object0->getpShape(i),
332  collision_object1->getpShape(j)))
333  {
334  // 情報の格納
335  information0.setOpponentCollisionBase(collision_object1->getpCollisionBase());
336  information1.setOpponentCollisionBase(collision_object0->getpCollisionBase());
337  information0.setMyObject(collision_object0);
338  information1.setMyObject(collision_object1);
339  information0.setOpponentObject(collision_object1);
340  information1.setOpponentObject(collision_object0);
341  information0.setMyShape(collision_object0->getpShape(i));
342  information1.setMyShape(collision_object1->getpShape(j));
343  information0.setOpponentShape(collision_object1->getpShape(j));
344  information1.setOpponentShape(collision_object0->getpShape(i));
345 
346  // 衝突応答
347  collision_object0->getpCollisionBase()->HitCollision(&information0);
348  collision_object1->getpCollisionBase()->HitCollision(&information1);
349  }
350  }
351  }
