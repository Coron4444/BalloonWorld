319 void CollisionManager::CollisionDetermination()
320 {
321  // ペアになっているため
322  unsigned max_index = collision_objects_list_.size() / 2;
323 
324  for (unsigned i = 0; i < max_index; i++)
325  {
326  // 複数衝突オブジェクトが判定するかどうか
327  if (!collision_objects_list_[i * 2]->getIsJudgment()) continue;
328  if (!collision_objects_list_[i * 2 + 1]->getIsJudgment()) continue;
329 
330  // 判定をする組み合わせか？
331  if (is_pair_check_)
332  {
333  if (!CollisionPairCheck::IsCheck(collision_objects_list_[i * 2]->getpCollisionBase(),
334  collision_objects_list_[i * 2 + 1]->getpCollisionBase())) continue;
335  }
336 
337  // 実際の判定処理
338  ActualCalculation(collision_objects_list_[i * 2],
339  collision_objects_list_[i * 2 + 1]);
340 
341  // 各コンポーネントの衝突オブジェクトの判定をリセット
342  if (collision_objects_list_[i * 2]->getIsJudgment())
343  {
344  collision_objects_list_[i * 2]->ResetHitDataAllCollisionObject();
345  }
346  if (collision_objects_list_[i * 2 + 1]->getIsJudgment())
347  {
348  collision_objects_list_[i * 2 + 1]->ResetHitDataAllCollisionObject();
349  }
350  }
351

359 void CollisionManager::OldCollisionDetermination()
360 {
361  // 衝突基底クラスが2つ以上かどうか
362  if (all_collision_.GetEndPointer() < 2) return;
363 
364  // 衝突判定
365  for (unsigned i = 0; i < all_collision_.GetEndPointer() - 1; i++)
366  {
367  // 衝突基底クラスがあるかどうか
368  if (all_collision_.GetArrayObject(i) == nullptr) continue;
369 
370  // 総当たり
371  for (unsigned j = i + 1; j < all_collision_.GetEndPointer(); j++)
372  {
373  // 衝突基底クラスがあるかどうか
374  if (all_collision_.GetArrayObject(j) == nullptr) continue;
375 
376  // 判定をする組み合わせか？
377  if (is_pair_check_)
378  {
379  if (!CollisionPairCheck::IsCheck(all_collision_.GetArrayObject(i),
380  all_collision_.GetArrayObject(j))) continue;
381  }
382 
383  for (unsigned a = 0; a < all_collision_.GetArrayObject(i)->getEndIndexOfArray(); a++)
384  {
385  // 複数衝突オブジェクトが判定するかどうか
386  if (!all_collision_.GetArrayObject(i)->getpCollisionObjects(a)
387  ->getIsJudgment()) continue;
388 
389  for (unsigned b = 0; b < all_collision_.GetArrayObject(i)->getEndIndexOfArray(); b++)
390  {
391  // 複数衝突オブジェクトが判定するかどうか
392  if (!all_collision_.GetArrayObject(j)->getpCollisionObjects(b)
393  ->getIsJudgment()) continue;
394 
395  // 実際の判定処理
396  ActualCalculation(all_collision_.GetArrayObject(i)->getpCollisionObjects(a),
397  all_collision_.GetArrayObject(j)->getpCollisionObjects(b));
398 
399  // 各コンポーネントの衝突オブジェクトの判定をリセット
400  if (all_collision_.GetArrayObject(i)->getpCollisionObjects(a)
401  ->getIsJudgment())
402  {
403  all_collision_.GetArrayObject(i)->getpCollisionObjects(a)
404  ->ResetHitDataAllCollisionObject();
405  }
406  else
407  {
408  a = all_collision_.GetArrayObject(i)->getEndIndexOfArray();
409  break;
410  }
411 
412  if (all_collision_.GetArrayObject(j)->getpCollisionObjects(b)
413  ->getIsJudgment())
414  {
415  all_collision_.GetArrayObject(j)->getpCollisionObjects(b)
416  ->ResetHitDataAllCollisionObject();
417  }
418  else
419  {
420  a = all_collision_.GetArrayObject(i)->getEndIndexOfArray();
421  break;
422  }
423 
424  }
425  }
426  }
427  }

435 void CollisionManager::CollisionGround()
436 {
437  if (ground_polygon_ == nullptr) return;
438 
439  for (unsigned i = 0; i < all_collision_.GetEndPointer(); i++)
440  {
441  // 地面との判定をするかどうか
442  if (!all_collision_.GetArrayObject(i)->getIsJudgmentGround()) continue;
443 
444  // 地面高さ取得
445  float position_y = ground_polygon_->GetHeight(*all_collision_.GetArrayObject(i)
446  ->getpGameObject()->GetTransform()
447  ->GetPosition());
448 
449  // 衝突してるか
450  if (position_y >= all_collision_.GetArrayObject(i)
451  ->getpGameObject()->GetTransform()->GetPosition()->y)
452  {
453  // 地面との衝突処理
454  all_collision_.GetArrayObject(i)->HitGround(position_y);
455  }
456  else
457  {
458  // 地面との非衝突処理
459  all_collision_.GetArrayObject(i)->NotHitGround(position_y);
460  }
461  }

468 //--------------------------------------------------
469 void CollisionManager::ActualCalculation(CollisionObjects* collision_objects0,
470  CollisionObjects* collision_objects1)
471 {
472  for (unsigned i = 0; i < collision_objects0->getEndIndexOfArray(); i++)
473  {
474  // 衝突オブジェクトが判定するかどうか
475  if (!collision_objects0->getpCollisionObject(i)->getIsJudgment()) continue;
476 
477  // 総当たり
478  for (unsigned j = 0; j < collision_objects1->getEndIndexOfArray(); j++)
479  {
480  // 衝突オブジェクトが判定するかどうか
481  if (!collision_objects1->getpCollisionObject(j)->getIsJudgment()) continue;
482 
483  // 衝突計算
484  if (SortCollisionCalculation(collision_objects0->getpCollisionObject(i),
485  collision_objects1->getpCollisionObject(j)))
486  {
487  // コンポーネント1つ目の衝突応答
488  collision_objects0->getpCollisionBase()
489  ->HitCollision(collision_objects1->getpCollisionBase(),
490  collision_objects1->getpCollisionObject(j),
491  collision_objects0->getpCollisionObject(i));
492 
493  // コンポーネント2つ目の衝突応答
494  collision_objects1->getpCollisionBase()
495  ->HitCollision(collision_objects0->getpCollisionBase(),
496  collision_objects0->getpCollisionObject(i),
497  collision_objects1->getpCollisionObject(j));
498  }
499  else
500  {
501  // コンポーネント1つ目の非衝突応答
502  collision_objects0->getpCollisionBase()
503  ->NotHitCollision(collision_objects1->getpCollisionBase(),
504  collision_objects1->getpCollisionObject(j),
505  collision_objects0->getpCollisionObject(i));
506 
507  // コンポーネント2つ目の非衝突応答
508  collision_objects1->getpCollisionBase()
509  ->NotHitCollision(collision_objects0->getpCollisionBase(),
510  collision_objects0->getpCollisionObject(i),
511  collision_objects1->getpCollisionObject(j));
512  }
513 
514  // コンポーネントが判定可能かどうか
515  if (!collision_objects0->getIsJudgment()) return;
516  if (!collision_objects1->getIsJudgment()) return;
517  }
518  }
519 }
