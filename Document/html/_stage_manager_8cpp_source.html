132 void StageManager::Update()
133 {
134  score_ += player_->getScore();
135 
136  // タイマー
137  time_--;
138  if (time_ < 0)
139  {
140  time_ = 0;
141  return;
142  }
143 
144  // 着地ボーナス
145  if (!player_->getIsOnTheGround()) return;
146  if (is_ground_) return;
147  is_ground_ = true;
148  score_ += BONUS_SCORE + BONUS_SCORE_RAITO * (int)TimeToFrame::FrameToSecond(time_);
149 }
