95 void GameObjectReferenceManager::AllReleaseReference()
96 {
97  // 参照元マップの解放
98  for(auto& contents : reference_map_.source_map_)
99  {
100  if (contents.second == nullptr) return;
101 
102  for (unsigned i = 0; i < contents.second->GetEndPointer(); i++)
103  {
104  ReferenceData* temp = contents.second->GetArrayObject(i);
105  SafeRelease::Normal(&temp);
106  }
107 
108  SafeRelease::Normal(&contents.second);
109  }
110 
111  // 参照先マップの解放
112  for(auto& contents : reference_map_.destination_map_)
113  {
114  if (contents.second == nullptr) return;
115 
116  for (unsigned i = 0; i < contents.second->GetEndPointer(); i++)
117  {
118  ReferenceData* temp = contents.second->GetArrayObject(i);
119  SafeRelease::Normal(&temp);
120  }
121 
122  SafeRelease::Normal(&contents.second);
123  }
124 }

134 void GameObjectReferenceManager::RegistrationReference_Source(GameObjectBase* source, void* pointer,
135  GameObjectBase* destination)
136 {
137  // 参照元の検索
138  auto iterator = reference_map_.source_map_.find(source);
139 
140  // 登録されていない場合
141  if (iterator == reference_map_.source_map_.end())
142  {
143  // 参照データ配列を作成
144  LimitedPointerArray<ReferenceData*, ARRAY_NUM>* temp_array = new LimitedPointerArray<ReferenceData*, ARRAY_NUM>;
145  temp_array->AddToArray(new ReferenceData(source, pointer, destination));
146 
147  // 参照元マップに参照データ配列を登録
148  reference_map_.source_map_.insert(std::make_pair(source, temp_array));
149  }
150  else
151  {
152  // 登録されている場合
153  unsigned index;
154  for (index = 0; index < iterator->second->GetEndPointer(); index++)
155  {
156  // 同じ参照先があるかどうか
157  if (iterator->second->GetArrayObject(index)->reference_destination_ != destination) continue;
158 
159  break;
160  }
161 
162  // 同じ参照先がないので登録
163  if (index != iterator->second->GetEndPointer())
164  {
165  // 追加
166  iterator->second->AddToArray(new ReferenceData(source, pointer, destination));
167  }
168  }
169 }

224 void GameObjectReferenceManager::ReleaseReference_Source(GameObjectBase* object)
225 {
226  auto iterator = reference_map_.source_map_.find(object);
227 
228  // 登録されている場合
229  if (iterator != reference_map_.source_map_.end())
230  {
231  for (unsigned i = 0; i < iterator->second->GetEndPointer(); i++)
232  {
233  // 解放
234  ReferenceData* temp = iterator->second->GetArrayObject(i);
235  SafeRelease::Normal(&temp);
236  }
237 
238  // 配列の解放
239  SafeRelease::Normal(&iterator->second);
240 
241  // 参照元マップから消去
242  reference_map_.source_map_.erase(object);
243  }
244 }
