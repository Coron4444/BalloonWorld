216 void EffekseerManager::AddShareData(const std::string* key_name, const std::string* file_path)
217 {
218  // 共有データマップにあるかの確認
219  auto itr_share = share_effekseer_map_.find(*key_name);
220  if (itr_share != share_effekseer_map_.end()) return;
221 
222  // 固有データマップにあるかの確認
223  auto itr_unique = unique_effekseer_map_.find(*key_name);
224  if (itr_unique != unique_effekseer_map_.end()) return;
225 
226  // 存在しない場合
227  std::string temp_path = MakeFilePath(key_name, file_path);
228  EffekseerArray* temp_array = new EffekseerArray(&temp_path);
229  share_effekseer_map_.insert(std::make_pair(*key_name, temp_array));
230  return;
231 }

315 void EffekseerManager::CreateProjectionMatrix(int angle_of_view_)
316 {
317  projection_matrix_ = Effekseer::Matrix44().PerspectiveFovLH(D3DXToRadian(angle_of_view_),
