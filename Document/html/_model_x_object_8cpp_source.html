149 void ModelXObject::InputMaterial(const std::string* file_path, LPD3DXBUFFER* material_buffer, bool is_share_data)
150 {
151  // マテリアルサイズ変更
152  material_.resize(material_num_);
153 
154  // マテリアルの情報をのポインタ取得
155  LPD3DXMATERIAL temp_material = (LPD3DXMATERIAL)(*material_buffer)->GetBufferPointer();
156 
157  // マテリアルで分割したパーツたち
158  for (DWORD i = 0; i < material_num_; i++)
159  {
160  ZeroMemory(&material_[i], sizeof(D3DMATERIAL9));
161 
162  // デヒューズカラーの設定
163  material_[i].Diffuse.r = temp_material[i].MatD3D.Diffuse.r;
164  material_[i].Diffuse.g = temp_material[i].MatD3D.Diffuse.g;
165  material_[i].Diffuse.b = temp_material[i].MatD3D.Diffuse.b;
166  material_[i].Diffuse.a = temp_material[i].MatD3D.Diffuse.a;
167 
168  // アンビエントカラーの設定
169  material_[i].Ambient.r = temp_material[i].MatD3D.Ambient.r;
170  material_[i].Ambient.g = temp_material[i].MatD3D.Ambient.g;
171  material_[i].Ambient.b = temp_material[i].MatD3D.Ambient.b;
172  material_[i].Ambient.a = temp_material[i].MatD3D.Ambient.a;
173 
174  // テクスチャの読み込み
175  if (temp_material[i].pTextureFilename == NULL) continue;
176  std::string texture_key_name = temp_material[i].pTextureFilename;
177  std::string texture_file_path = *file_path;
178  MakeTextureFilePathAndKeyName(&texture_key_name, &texture_file_path);
179 
180  TextureObject* texture_object = nullptr;
181 
182  if (is_share_data)
183  {
184  texture_object = TextureManager::AddShareData(&texture_key_name, &texture_file_path);
185  }
186  else
187  {
188  texture_object = TextureManager::AddUniqueData(&texture_key_name, &texture_file_path);
189  }
190 
191  texture_object_array_.push_back(texture_object);
192  }
193 
194  // マテリアルバッファの解放
195  SafeRelease::PlusRelease(material_buffer);
196 }

206 void ModelXObject::MakeTextureFilePathAndKeyName(std::string* texture_key_name, std::string* texture_file_path)
207 {
208  // ファイルパスからフォルダ名を/付きで取得
209  auto slash_index = texture_file_path->rfind("/");

211  unsigned temp_num = (unsigned)(slash_index - slash_index2);
212  std::string temp_strign = texture_file_path->substr(slash_index2 + 1, temp_num);
213 
214  // キーネームの作成
215  *texture_key_name = temp_strign + *texture_key_name;
216 
217  // ファイルパスの作成
218  *texture_file_path = texture_file_path->substr(0, slash_index2 + 1);
219 }
