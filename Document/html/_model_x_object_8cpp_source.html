164 void ModelXObject::CreateMaterial(std::string* file_path, LPD3DXBUFFER* material_buffer)
165 {
166  // マテリアルサイズ変更
167  material_.resize(mesh_num_);
168 
169  // マテリアルの情報をのポインタ取得
170  LPD3DXMATERIAL temp_material = (LPD3DXMATERIAL)(*material_buffer)->GetBufferPointer();
171 
172  // マテリアルで分割したパーツたち
173  for (DWORD i = 0; i < mesh_num_; i++)
174  {
175  ZeroMemory(&material_[i], sizeof(D3DMATERIAL9));
176 
177  // デヒューズカラーの設定
178  material_[i].Diffuse.r = temp_material[i].MatD3D.Diffuse.r;
179  material_[i].Diffuse.g = temp_material[i].MatD3D.Diffuse.g;
180  material_[i].Diffuse.b = temp_material[i].MatD3D.Diffuse.b;
181  material_[i].Diffuse.a = temp_material[i].MatD3D.Diffuse.a;
182 
183  // アンビエントカラーの設定
184  material_[i].Ambient.r = temp_material[i].MatD3D.Ambient.r;
185  material_[i].Ambient.g = temp_material[i].MatD3D.Ambient.g;
186  material_[i].Ambient.b = temp_material[i].MatD3D.Ambient.b;
187  material_[i].Ambient.a = temp_material[i].MatD3D.Ambient.a;
188 
189  // テクスチャの読み込み
190  if (temp_material[i].pTextureFilename == NULL) continue;
191  std::string key_name = temp_material[i].pTextureFilename;
192  std::string path = *file_path;
193  CreateFilePathAndKeyName(&path, &key_name);
194 
195  TextureObject* texture_object = TextureManager::getpInstance()->getpObject(&key_name, &path);
196  diffuse_texture_object_.push_back(texture_object);
197  }
198 
199  // マテリアルバッファの解放
200  SafeRelease::PlusRelease(material_buffer);
201 }
