254 void NumbersPolygon::UpdateEachDigit(int number, int digits_num)
255 {
256  // ずらし量
257  float AmountOfShift = (number_.size() - 1) * (scale_.x / 2);
258 
259  for (int i = (digits_num - 1); i >= 0; i--)
260  {
261  int temp_number = number % 10;
262  number /= 10;
263 
264  number_[i]->setNumber(temp_number);
265  number_[i]->setColor(color_);
266  *number_[i]->getpTransform()->getpPosition() = Vec3(position_.x + AmountOfShift,
267  position_.y, 0.0f);
268  *number_[i]->getpTransform()->getpScale() = Vec3(scale_.x, scale_.y, 1.0f);
269  number_[i]->getpTransform()->CreateWorldMatrix();
270 
271  AmountOfShift -= (scale_.x / 2);
272 
273  // ゼロ詰めしないならループ終了
274  if (!is_zero_ && !number) return;
275  }
276 }
