71 void UpdateManager::Update()
72 {
73  // 追加待ち配列の中身を追加
74  AddContentsOfAwaitAddArray();
75 
76  // 解放待ち配列の中身を解放
77  ReleaseContentsOfAwaitReleaseArray();
78 
79  // 更新
80  for (unsigned i = 0; i < all_update_.GetEndPointer(); i++)
81  {
82  if (all_update_.GetArrayObject(i) == nullptr) continue;
83  all_update_.GetArrayObject(i)->Update();
84  }
85 
86  // 後更新
87  for (unsigned i = 0; i < all_update_.GetEndPointer(); i++)
88  {
89  if (all_update_.GetArrayObject(i) == nullptr) continue;
90  all_update_.GetArrayObject(i)->LateUpdate();
91  }
92 
93  // 物理更新
94  for (unsigned i = 0; i < all_update_.GetEndPointer(); i++)
95  {
96  if (all_update_.GetArrayObject(i)->getpGameObject()->GetPhysics() == nullptr) continue;
97  all_update_.GetArrayObject(i)->getpGameObject()->GetPhysics()->Update();
98  }
99 
100 
101  // デバッグ
102 #ifdef _DEBUG
103  for (unsigned i = 0; i < all_update_.GetEndPointer(); i++)
104  {
105  if (all_update_.GetArrayObject(i) == nullptr) continue;
106  all_update_.GetArrayObject(i)->DebugDisplay();
107  }
108 #endif
109 }

127 void UpdateManager::OverwriteArrayUpdateBase(GameObjectBase* game_object,
128  UpdateBase* new_update)
129 {
130  // 新規がnullptrの場合
131  if (new_update == nullptr)
132  {
133  // 古い更新基底クラスの解放
134  ReleaseUpdateBaseFromArray(game_object->GetUpdate());
135 
136  // 古い更新基底クラスの消去
137  UpdateBase* temp = game_object->GetUpdate();
138  SafeRelease::Normal(&temp);
139 
140  // nullptrの代入
141  game_object->SetUpdate(new_update);
142  }
143  else
144  {
145  // 配列の上書き
146  all_update_.OverwriteArray(game_object->GetUpdate(), new_update);
147 
148  // 古い更新基底クラスの消去
149  UpdateBase* temp = game_object->GetUpdate();
150  SafeRelease::Normal(&temp);
151 
152  // 新規更新基底クラスの初期化
153  game_object->SetUpdate(new_update);
154  *new_update->getpGameObject() = *game_object;
155  new_update->Init();
156  }
157 }
