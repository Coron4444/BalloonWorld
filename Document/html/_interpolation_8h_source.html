419

461  if (interpolation_->size() <= 1) return 1;
462 
463  // 距離を加算
464  current_distance_ += distance;
465  cumulative_distance_ += distance; // 累積にも追加
466 
467  // 現在の目標要素を確定
468  for (; ;)
469  {
470  // 目標要素が存在するか確認
471  if (target_index_ >= max_index_) return 1;
472 
473  // 現在の距離が目標要素区間幅を超えていないか確認
474  if (current_distance_
475  >= *first_pointer_[offset_index_ + target_index_]->getpDistance())
476  {
477  // 飛び出た距離を算出し目標要素インデックスを次に変更する
478  current_distance_ -= *first_pointer_[offset_index_ + target_index_]->getpDistance();
479  target_index_++;
480  continue;
481  }
482  break;
483  }
484  target_element_ = first_pointer_[offset_index_ + target_index_];
485 
486  // 目標要素に対する補間距離の算出(0~1)
487  return current_distance_ / *target_element_->getpDistance();

504  memcpy(element_array, first_pointer_ + target_index_ - 1,
505  need_element_num * sizeof(Element*));
506  return true;
