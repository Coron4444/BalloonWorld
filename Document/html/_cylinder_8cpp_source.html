111  right.ChangeAnyLength(radius_);
112 
113  Vector3D temp_max = *getpPosition() + up + right;
114  max.x = temp_max.x;

117 
118  Vector3D temp_min = *getpPosition() + -right;
119  min.x = temp_min.x;
120  temp_min = *getpPosition() + right;
121  min.y = temp_min.y;
122 
123  // Z座標
124  axis.Reset();
125  axis.RotationAxisX(D3DXToRadian(45.0f));
126  up = *axis.getpUp();
127  Vector3D forward = *axis.getpForward();
128  up.ChangeAnyLength(getpVector()->getLength());
129  forward.ChangeAnyLength(radius_);
130 
131  temp_max = *getpPosition() + up + forward;
132  max.z = temp_max.z;
133 
134  temp_min = *getpPosition() + -forward;
135  min.z = temp_min.z;
136 
137  *getpMax() = max;
