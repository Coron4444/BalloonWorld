191 void MeshPlanePolygon::CreateVertex(Vec2 mesh_num, Vec2 mesh_scale, Color4 color)
192 {
193  // 頂点数の決定
194  int temp_vertex_num = 4 + 2 * (((int)mesh_num.x) - 1);
195  temp_vertex_num += temp_vertex_num / 2 * (((int)mesh_num.y) - 1);
196  vertex_.resize(temp_vertex_num);
197 
198  // 頂点情報の入力
199  float coordinate_x = -((mesh_scale.x / 2) * ((int)mesh_num.x));
200  float coordinate_z = ((mesh_scale.y / 2) * ((int)mesh_num.y));
201  float tex_widht = 0.0f;
202  float tex_heihgt = 0.0f;
203  int x_count = 0;
204 
205  for (std::vector<RendererDirectX9::VERTEX_3D>::size_type i = 0; i < vertex_.size(); i++)
206  {
207  vertex_[i].posisiont_ = Vec3(coordinate_x, 0.0f,/*field_height[i],*/ coordinate_z);
208  vertex_[i].color_ = color;
209  vertex_[i].texcoord_ = Vec2(tex_widht, tex_heihgt);
210  coordinate_x += mesh_scale.x;
211  tex_widht += TEXTURE_WIDTH;
212 
213  bool is_x_end = x_count == ((int)mesh_num.x);
214  if (is_x_end)
215  {
216  x_count = 0;
217  coordinate_x = -((mesh_scale.x / 2) * ((int)mesh_num.x));
218  tex_widht = 0.0f;
219  coordinate_z += -mesh_scale.y;
220  tex_heihgt += TEXTURE_HEIGHT;
221  }
222  else
223  {
224  x_count++;
225  }
226  }
227 
228 
229  // 法線計算
230  for (std::vector<RendererDirectX9::VERTEX_3D>::size_type i = 0; i < vertex_.size(); i++)
231  {
232  // フィールドの後ろ端の頂点
233  if (i / (unsigned)(mesh_num.x + 1.0f) == 0)
234  {
235  vertex_[i].normal_ = Vec3(0.0f, 1.0f, 0.0f);
236  continue;
237  }
238 
239  // フィールドの前端の頂点
240  if (i / (unsigned)(mesh_num.x + 1.0f) == (unsigned)mesh_num.y)
241  {
242  vertex_[i].normal_ = Vec3(0.0f, 1.0f, 0.0f);
243  continue;
244  }
245 
246  // フィールドの左端の頂点
247  if (i % (unsigned)(mesh_num.x + 1.0f) == 0)
248  {
249  vertex_[i].normal_ = Vec3(0.0f, 1.0f, 0.0f);
250  continue;
251  }
252 
253  // フィールドの右端の頂点
254  if (i % (unsigned)(mesh_num.x + 1.0f) == (unsigned)mesh_num.x)
255  {
256  vertex_[i].normal_ = Vec3(0.0f, 1.0f, 0.0f);
257  continue;
258  }
259 
260  // それ以外の頂点
261  // x方向
262  Vec3 temp_vector = vertex_[i + 1].posisiont_ - vertex_[i - 1].posisiont_;
263  Vec3 normal_x = Vec3(-temp_vector.y, temp_vector.x, 0.0f);
264 
265  // z方向
266  temp_vector = vertex_[i - 5].posisiont_ - vertex_[i + 5].posisiont_;
267  Vec3 normal_z = Vec3(0.0f, temp_vector.z, -temp_vector.y);
268  Vector3D normal = normal_x + normal_z;
269  vertex_[i].normal_ = *normal.ChangeNormalize();
270  }

430 bool MeshPlanePolygon::CalculationHeight(Vector3D* position, int index0,
431  int index1, int index2)
432 {
433  // 三角形の各辺に沿うベクトルを算出
434  Vector3D v01 = vertex_[index1].posisiont_ - vertex_[index0].posisiont_;
435  Vector3D v12 = vertex_[index2].posisiont_ - vertex_[index1].posisiont_;
436  Vector3D v20 = vertex_[index0].posisiont_ - vertex_[index2].posisiont_;
437 
438  // 各頂点から自分へのベクトル
439  Vector3D v0p = *position - vertex_[index0].posisiont_;
440  Vector3D v1p = *position - vertex_[index1].posisiont_;
441  Vector3D v2p = *position - vertex_[index2].posisiont_;
442 
443  // 上記2つの外積を求める(2D外積)
444  float c0 = (v01.x * v0p.z) - (v01.z * v0p.x);
445  float c1 = (v12.x * v1p.z) - (v12.z * v1p.x);
446  float c2 = (v20.x * v2p.z) - (v20.z * v2p.x);
447 
448  // 自分がポリゴン内にいる場合
449  if (c0 <= 0.0f && c1 <= 0.0f && c2 <= 0.0f)
450  {
451  // y座標を求める
452 
453  Vector3D temp_normal = Vector3D::CreateCross(&v01, &v12);
454  Vector3D p0 = vertex_[index0].posisiont_;
455  position->y = p0.y - ((temp_normal.x * (position->x - p0.x) +
456  temp_normal.z * (position->z - p0.z)) / temp_normal.y);
457  return true;
458  }
459  return false;
460 }
