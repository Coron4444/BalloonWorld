834 bool CollisionCalculation::QuadraticEquationCalculationForLineSegmentsAndSpheres(QuadraticEquation* equation,
835  LineSegment* line_segment,

863 bool CollisionCalculation::QuadraticEquationCalculationForLineSegmentsAndCylinders(QuadraticEquation* equation,
864  LineSegment* line_segment,
865  Cylinder* cylinder)
