133 {

136  {
137  case PlayerCollision::ObjectTag::BOUNDING_SPHERE:

139  HitBoundingSphere(hit_collision, hit_object, hit_my_object);
140  break;
141  }
142  case PlayerCollision::ObjectTag::SUBSTANCE:
143  {
144  HitSubstance(hit_collision, hit_object, hit_my_object);
145  break;
146  }
