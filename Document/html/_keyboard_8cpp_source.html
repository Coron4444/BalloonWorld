116 void Keyboard::Update()
117 {
118  BYTE temp_key_state[MAX_KEY_NUM];
119 
120  // デバイスからデータを取得
121  if (SUCCEEDED(keyboard_->GetDeviceState(sizeof(temp_key_state), temp_key_state)))
122  {
123  for (int i = 0; i < MAX_KEY_NUM; i++)
124  {
125  // キートリガー・リリース情報を生成
126  key_state_trigger_[i] = (key_state_[i] ^ temp_key_state[i]) & temp_key_state[i];
127  key_state_release_[i] = (key_state_[i] ^ temp_key_state[i]) & key_state_[i];
128 
129 
130  // キーリピート情報を生成
131  if (temp_key_state[i])
132  {
133  if (repeat_counter_[i] < LIMIT_COUNT_REPEAT)
134  {
135  repeat_counter_[i]++;
136  if (repeat_counter_[i] == 1
137  || repeat_counter_[i] >= LIMIT_COUNT_REPEAT)
138  {// キーを押し始めた最初のフレーム、または一定時間経過したらキーリピート情報ON
139  key_state_repeat_[i] = temp_key_state[i];
140  }
141  else
142  {
143  key_state_repeat_[i] = 0;
144  }
145  }
146  }
147  else
148  {
149  repeat_counter_[i] = 0;
150  key_state_repeat_[i] = 0;
151  }
152 
153  // キープレス情報を保存
154  key_state_[i] = temp_key_state[i];
155  }
156  }
157  else
158  {
159  // キーボードへのアクセス権を取得
160  keyboard_->Acquire();
161  }
162 }
