181 void CollisionObject::UpdateAABB()
182 {
183  Vector3D max;
184  Vector3D min;
185  for (unsigned i = 0; i < all_shape_.getEndIndex(); i++)
186  {
187  Vector3D temp_max = *all_shape_.getObject(i)->getpShape()->getpMax();
188  Vector3D temp_min = *all_shape_.getObject(i)->getpShape()->getpMin();
189 
190  // 最小が上、最大が下の場合
191  if (temp_min.y >= temp_max.y)
192  {
193  temp_max.y = -temp_max.y;
194  temp_min.y = -temp_min.y;
195  }
196  // 最小が奥、最大が手前の場合
197  if (temp_min.z >= temp_max.z)
198  {
199  temp_max.z = -temp_max.z;
200  temp_min.z = -temp_min.z;
201  }
202 
203  if (max.x < temp_max.x) max.x = temp_max.x;
204  if (max.y < temp_max.y) max.y = temp_max.y;
205  if (max.z < temp_max.z) max.z = temp_max.z;
206 
207  if (min.x > temp_min.x) min.x = temp_min.x;
208  if (min.y > temp_min.y) min.y = temp_min.y;
209  if (min.z > temp_min.z) min.z = temp_min.z;
210  }
211 
212  *octree_aabb_.getpMax() = max;
213  *octree_aabb_.getpMin() = min;
214 }
