209 void CubePolygon::AccessVRAM()
210 {
211  // VRAMのメモリを確保(GPUに依頼)(頂点バッファの作成)
212  HRESULT hr = device_->CreateVertexBuffer(sizeof(RendererDirectX9::VERTEX_3D) * (int)vertex_.size(), // 借りたいbafの量(バイト)、つまり1頂点の容量×必要頂点数
213  D3DUSAGE_WRITEONLY, // 使用用途(今回は書き込みのみ、GPUが早く動くが書き込んだデータを読んではダメ(値が不定))
214  RendererDirectX9::FVF_VERTEX_3D, // 頂点属性
215  D3DPOOL_MANAGED, // 頂点バッファの管理方法( MANAGEDは管理はDirect3Dにお任せという意味 )
216  &vertex_buffer_, // 管理者の居場所のメモ帳(ポインタのポインタ)(全てはこれの値を知るための作業)
217  NULL); // なぞ？
218 
219  if (FAILED(hr))
220  {
221  MessageBox(NULL, "Vertex buffer was not created.", "Error", MB_OK);
222  }
223 
224 
225  // VRAMのメモリを確保(GPUに依頼)(インデックスバッファの作成)
226  hr = device_->CreateIndexBuffer(sizeof(WORD) * (int)index_.size(), // 借りたいbafの量(バイト)、つまり1頂点の容量×必要頂点数
227  D3DUSAGE_WRITEONLY, // 使用用途(今回は書き込みのみ、GPUが早く動くが書き込んだデータを読んではダメ(値が不定))
228  D3DFMT_INDEX16, // 頂点フォーマット(WORD型だから16、DWORD型なら32)
229  D3DPOOL_MANAGED, // 頂点バッファの管理方法( MANAGEDは管理はDirect3Dにお任せという意味 )
230  &index_buffer_, // 管理者の居場所のメモ帳(ポインタのポインタ)(全てはこれの値を知るための作業)
231  NULL); // なぞ？
232 
233  if (FAILED(hr))
234  {
235  MessageBox(NULL, "Index buffer was not created.", "Error", MB_OK);
236  }
237 
238  RegistrationVertex();
239  RegistrationIndex();
240 }
