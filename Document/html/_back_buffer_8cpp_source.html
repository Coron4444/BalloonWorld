382 void BackBuffer::SortTransparent()
383 {
384  // 透明オブジェクトがあるかどうか
385  if (all_transparency_draw_.GetEndPointer() <= 0) return;
386 
387  for (unsigned i = 0; i < all_transparency_draw_.GetEndPointer() - 1; i++)
388  {
389  // 深度値を算出
390  Vector3D temp_vector0 = *all_transparency_draw_.GetArrayObject(i)->getpGameObject()
391  ->GetTransform()->GetPosition() - *camera_->GetPositon();
392 
393  float depth_value0 = temp_vector0.GetLengthSquare();
394 
395  for (unsigned j = i + 1; j < all_transparency_draw_.GetEndPointer(); j++)
396  {
397  // 深度値を算出
398  Vector3D temp_vector1 = *all_transparency_draw_.GetArrayObject(j)
399  ->getpGameObject()->GetTransform()->GetPosition() - *camera_->GetPositon();
400 
401  float depth_value1 = temp_vector1.GetLengthSquare();
402 
403  // 深度値を比較
404  if (depth_value0 < depth_value1)
405  {
406  // 並び替え
407  all_transparency_draw_.SortTheTwoObject(i, j);
408  }
409  }
410  }
411 }
