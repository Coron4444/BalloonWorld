174 void BulletPhysicsObject::InitCommon(float mass, Vector3D* inertia,
175  Vector3D* position, Quaternion* quaternion)
176 {
177  // 慣性モーメント算出
178  btVector3 temp_inertia = btVector3(inertia->x, inertia->y, inertia->z);
179  collision_shape_->calculateLocalInertia(mass, temp_inertia);
180 
181  // 外部から剛体を操作するハンドル作成
182  btTransform transform;
183  transform.setIdentity();
184  transform.setOrigin(btVector3(position->x, position->y, position->z));
185  transform.setRotation(btQuaternion(quaternion->x, quaternion->y,
186  quaternion->z, quaternion->w));
187  motion_state_ = new btDefaultMotionState(transform);
188 
189  // 剛体作成
190  btRigidBody::btRigidBodyConstructionInfo rigidbody_info(mass, motion_state_,
191  collision_shape_,
192  temp_inertia);
193  rigid_body_ = new btRigidBody(rigidbody_info);
194  rigid_body_->setAnisotropicFriction(collision_shape_
195  ->getAnisotropicRollingFrictionDirection(),
196  btCollisionObject::
197  CF_ANISOTROPIC_ROLLING_FRICTION);
198 }
