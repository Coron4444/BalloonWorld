90 void EffekseerObject::Update()
91 {
92  // ワールド行列の設定
93  SetWorldMatrix(transform_.getpWorldMatrix());
94 
95  // 再生中かどうか
96  is_playing_ = effekseer_manager_->Exists(effekseer_handle_);
97 
98  if (!is_playing_)
99  {
100  // リピートするかどうか
101  if (is_repeat_)
102  {
103  Play();
104  }
105  else
106  {
107  Stop();
108  }
109  }
110 
111  effekseer_manager_->Flip();
112 
113  //すべてのエフェクトの更新
114  effekseer_manager_->Update();
115 
116 }

239 void EffekseerObject::InitRenderer_Manager()
240 {
241  // デバイスの取得
242  LPDIRECT3DDEVICE9 device = nullptr;
243  Renderer::getpInstance()->getDevice(&device);
244  if (device == nullptr)
245  {
246  MessageBox(nullptr, "NotGetDevice!(EffekseerObject.cpp)", "Error", MB_OK);
247  return;
248  }
249 
250  // エフェクシアの初期化
251  effekseer_renderer_ = EffekseerRendererDX9::Renderer::Create(device, MAX_SPRITE_NUM);
252  effekseer_manager_ = Effekseer::Manager::Create(MAX_SPRITE_NUM);
253 
254  // 描画機能の設定
255  effekseer_manager_->SetSpriteRenderer(effekseer_renderer_->CreateSpriteRenderer());
256  effekseer_manager_->SetRibbonRenderer(effekseer_renderer_->CreateRibbonRenderer());
257  effekseer_manager_->SetRingRenderer (effekseer_renderer_->CreateRingRenderer());
258  effekseer_manager_->SetTrackRenderer (effekseer_renderer_->CreateTrackRenderer());
259  effekseer_manager_->SetModelRenderer (effekseer_renderer_->CreateModelRenderer());
260 
261  // テクスチャの読み込みを設定(今回はファイルから読み込み)
262  effekseer_manager_->SetTextureLoader(effekseer_renderer_->CreateTextureLoader());
263  effekseer_manager_->SetModelLoader (effekseer_renderer_->CreateModelLoader());
264 
265  // 左手座標系に設定
266  effekseer_manager_->SetCoordinateSystem(Effekseer::CoordinateSystem::LH);
267 }
