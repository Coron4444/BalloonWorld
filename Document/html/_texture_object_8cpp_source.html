62 void TextureObject::setDivisionNum(int division_num, int division_num_width,
63  int division_num_height)
64 {
65  // 切り取り幅高さ算出
66  float cut_width = ((float)width_ / division_num_width) / (float)width_;
67  float cut_height = ((float)height_ / division_num_height) / (float)height_;
68 
69  // UVの作成
70  uv_.clear();
71  division_num_ = division_num;
72  for (int i = 0; i < division_num; i++)
73  {
74  int animation_num_width = i % division_num_width;
75  int animation_num_height = i / division_num_height;
76 
77  UV uv;
78  *uv.getpLeftUpU() = animation_num_width * cut_width;
79  *uv.getpLeftUpV() = animation_num_height * cut_height;
80  *uv.getpRightButtomU() = *uv.getpLeftUpU() + cut_width;
81  *uv.getpRightButtomV() = *uv.getpLeftUpV() + cut_height;
82 
83  uv_.push_back(uv);
84  }
85 }
