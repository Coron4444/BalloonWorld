113 void TextureObject::SetTextureInformation(const std::string* file_path, const int* pattern_num_width, const int* pattern_num_height)
114 {
115  // テクスチャファイルから情報を取り出す
116  D3DXIMAGE_INFO image_info;
117  D3DXGetImageInfoFromFile(file_path->c_str(), &image_info);
118 
119  width_ = (int)image_info.Width;
120  height_ = (int)image_info.Height;
121 
122  // UV値の算出
123  float tcw = ((float)width_ / *pattern_num_width ) / (float)width_;
124  float tch = ((float)height_ / *pattern_num_height) / (float)height_;
125 
126  for ( int i = 0; i < pattern_num_all_; i++)
127  {
128  int pattern_width = i % *pattern_num_width;
129  int pattern_height = i / *pattern_num_width;
130 
131  TextureUV temp_uv;
132  temp_uv.u0 = pattern_width * tcw;
133  temp_uv.v0 = pattern_height * tch;
134  temp_uv.u1 = temp_uv.u0 + tcw;
135  temp_uv.v1 = temp_uv.v0 + tch;
136 
137  texture_uv.push_back(temp_uv);
138  }
139 }
