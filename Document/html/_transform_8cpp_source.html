88 void Transform::SetLookAtVector(Vector3D look_at_vector)
89 {
90  // 入力ベクトルの正規化
91  look_at_vector.CreateNormalize();
92 
93  // 初期状態の前ベクトルと見る方向との外積を算出
94  Vector3D temp_vector = axis_vector_.GetDefaultForawrd()->CreateVectorCross(look_at_vector);
95 
96  // ターゲットクォータニオンの回転角度の算出
97  float temp_dot = axis_vector_.GetDefaultForawrd()->CreateVectorDot(look_at_vector);
98 
99  // 外積に使用したベクトルが平行だった場合
100  if ((int)temp_vector.GetLengthSquare() == 0)
101  {
102  // 更にベクトルの向きが真逆だった場合
103  if (temp_dot < 0)
104  {
105  // 初期値のクォータニオンと真逆のターゲットクォータニオンの作成
106  SetTargetQuaternion(*axis_vector_.GetDefaultUp(), D3DXToRadian(180.0f));
107  D3DXQuaternionInverse(&target_quaternion_, &target_quaternion_);
108  return;
109  }
110  }
111 
112 
113  // ターゲットクォータニオンの作成
114  temp_vector.CreateNormalize();
115  SetTargetQuaternion(temp_vector, acosf(temp_dot));
116 }

241 void Transform::CreateQuaternion()
242 {
243  // YawPitchRollクォータニオンの作成
244  D3DXQuaternionIdentity(&yaw_pitch_roll_quaternion_);
245  D3DXQuaternionRotationYawPitchRoll(&yaw_pitch_roll_quaternion_, yaw_pitch_roll_.y, yaw_pitch_roll_.x, yaw_pitch_roll_.z);
246 
247  // 球面線形補完によるクォータニオンの作成
248  look_at_change_amount_ += look_at_speed_;
249  if (look_at_change_amount_ >= 1.0f) look_at_change_amount_ = 1.0f;
250  D3DXQuaternionSlerp(&now_quaternion_, &old_target_quaternion_, &target_quaternion_, look_at_change_amount_);
251 
252  // クォータニオンの結合
253  D3DXQuaternionMultiply(&now_quaternion_, &now_quaternion_, &yaw_pitch_roll_quaternion_);
254 }
