32 template<class Type, unsigned ARRAY_NUM = 100000>
33 class LimitedPointerArray

40  {
41  // 配列の初期化
42  for (unsigned i = 0; i < ARRAY_NUM; i++)
43  {
44  array_[i] = nullptr;
45  }

55  // 配列に追加
56  void AddToArray(Type object)
57  {
58  // 登録の確認
59  auto iterator = registration_map_.find(object);
60  if (iterator != registration_map_.end()) return;
61 
62  // 追加
63  for (unsigned i = 0; i < ARRAY_NUM; i++)
64  {
65  if (array_[i] != nullptr) continue;
66 
67  // 配列に追加
68  array_[i] = object;
69 
70  // マップに登録
71  registration_map_.insert(std::make_pair(object, i));
72 
73  // エンドポインタの更新
74  if (end_pointer_ < (i + 1))
75  {
76  end_pointer_ = (i + 1);
77  }
78  return;
79  }
80  }

84  // 配列の上書き
85  void OverwriteArray(Type old_object, Type new_object)
86  {
87  // 登録の確認
88  auto iterator = registration_map_.find(old_object);
89  if (iterator != registration_map_.end())
90  {
91  // 配列の上書き
92  array_[iterator->second] = new_object;
93 
94  // 新規オブジェクトを登録及び古いオブジェクトの削除
95  registration_map_.insert(std::make_pair(new_object, iterator->second));
96  registration_map_.erase(old_object);
97  }
98  else
99  {
100  // 古いオブジェクトがないため新規扱い
101  AddToArray(new_object);
102  }
103  }

178  // 配列のソート
179  void SortArray()
180  {
181  int temp_num = 0;
182 
183  for (unsigned i = 0; i < end_pointer_; i++)
184  {
185  if (array_[i] != nullptr) continue;
186 
187  temp_num++;
188 
189  for (unsigned k = (end_pointer_ - 1); k > i; k--)
190  {
191  if (array_[k] == nullptr) continue;
192 
193  // オブジェクトを入れ替える
194  array_[i] = array_[k];
195  array_[k] = nullptr;
196 
197  // マップの更新
198  registration_map_.find(array_[i])->second = i;
199  break;
200  }
201  }
202 
203  // エンドポインタの更新
204  end_pointer_ -= temp_num;
205  }

212  if (iterator == registration_map_.end()) return;
213

247  {

251

254  assert(array_[index1] != nullptr && "配列がnullptrです(LimitedPointerArray.h)");
255 
256  // マップの更新

258  registration_map_.find(array_[index0])->second = registration_map_.find(array_[index1])->second;
259  registration_map_.find(array_[index1])->second = temp_index;
260
