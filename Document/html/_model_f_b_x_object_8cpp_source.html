120 void ModelFBXObject::CreateVertex(mfmlib::Model* temp_model, std::vector<Shaderbuff>* temp_mesh, const std::string* file_path)
121 {
122  // バッファを総メッシュ分増やす
123  temp_mesh->resize(temp_model->MeshNum);
124  mfm_model_data_.MeshData.resize(temp_model->MeshNum);
125 
126  // メッシュ数分回す
127  for( unsigned i = 0; i < temp_mesh->size(); i++ )
128  {
129  // インデックス数分回す
130  temp_mesh->at(i).Vertex.resize(temp_model->MeshData[i].Index.Num);
131 
132  // インデックス番号の取得
133  mfm_model_data_.MeshData[i].MaterialIndex = temp_model->MeshData[i].MaterialIndex;
134 
135  // ？
136  temp_model->BoneData[i].BoneNum;
137 
138  // 頂点情報取得
139  for (unsigned j = 0; j < temp_mesh->at(i).Vertex.size(); j++)
140  {
141  // 頂点
142  int temp_index_num = temp_model->MeshData[i].Index.Data[j];
143  temp_mesh->at(i).Vertex[j].Pos.x = temp_model->MeshData[i].Vertex.Data[temp_index_num].x;
144  temp_mesh->at(i).Vertex[j].Pos.y = temp_model->MeshData[i].Vertex.Data[temp_index_num].y;
145  temp_mesh->at(i).Vertex[j].Pos.z = temp_model->MeshData[i].Vertex.Data[temp_index_num].z;
146 
147  // 法線
148  temp_mesh->at(i).Vertex[j].Normal.x = temp_model->MeshData[i].Normals.Data[j].x;
149  temp_mesh->at(i).Vertex[j].Normal.y = temp_model->MeshData[i].Normals.Data[j].y;
150  temp_mesh->at(i).Vertex[j].Normal.z = temp_model->MeshData[i].Normals.Data[j].z;
151 
152  // UV
153  if ( temp_model->MeshData[ i ].TexUV.Num != 0 )
154  {
155  temp_mesh->at(i).Vertex[j].texcoord.x = temp_model->MeshData[i].TexUV.Data[j].U;
156  temp_mesh->at(i).Vertex[j].texcoord.y = temp_model->MeshData[i].TexUV.Data[j].V;
157  }
158  else
159  {
160  temp_mesh->at(i).Vertex[j].texcoord.x = 0.0f;
161  temp_mesh->at(i).Vertex[j].texcoord.y = 0.0f;
162  }
163 
164  // カラー
165  XColor4 temp_color;
166  temp_color.r = temp_model->MaterialData[mfm_model_data_.MeshData[i].MaterialIndex].Color.Diffuse.x;
167  temp_color.g = temp_model->MaterialData[mfm_model_data_.MeshData[i].MaterialIndex].Color.Diffuse.y;
168  temp_color.b = temp_model->MaterialData[mfm_model_data_.MeshData[i].MaterialIndex].Color.Diffuse.z;
169 
170  if (temp_model->MaterialData[mfm_model_data_.MeshData[i].MaterialIndex].Color.Transparency == 0.0f)
171  {
172  temp_color.a = 1.0f;
173  }
174  else
175  {
176  temp_color.a = temp_model->MaterialData[mfm_model_data_.MeshData[i].MaterialIndex].Color.Transparency;
177  }
178 
179  temp_mesh->at(i).Vertex[j].color = (Color4)temp_color;
180 
181 
182  // ボーンの重み
183  if (temp_model->BoneData[i].BoneNum > 0)
184  {
185  for (int k = 0; k < MAX_MFM_BONE; k++)
186  {
187  // 頂点へのボーンの重みとボーン番号取得
188  temp_mesh->at(i).Vertex[j].ImpactMap.weight[k] = temp_model->BoneData[i].VertexData.Data[temp_index_num].weight[k];
189  temp_mesh->at(i).Vertex[j].ImpactMap.matrixIndex[k] = (unsigned char)temp_model->BoneData[i].VertexData.Data[temp_index_num].index[k];
190  }
191  }
192  }
193 
194 
195  // テクスチャ名の読み込み
196  std::string temp_name = temp_model->MaterialData[mfm_model_data_.MeshData[i].MaterialIndex].pTexName;
197  if(temp_name != "")
198  {
199  mfm_model_data_.MeshData[i].TexName = temp_name;
200  texture_manager_->RegistrationTextureObject(&mfm_model_data_.MeshData[i].TexName, file_path);
201  }
202 
203  // プリミティブ数の取得
204  mfm_model_data_.MeshData[i].PrimitiveNum = temp_model->MeshData[i].PrimitiveIndex;
205 
206  // アニメーションフレーム数の取得
207  mfm_model_data_.nAnimFrame = temp_model->FrameNum;
208 
209  // メッシュに含まれるボーンの取得
210  mfm_model_data_.bone.resize(temp_mesh->size()); // メッシュ数分ボーン領域確保
211  mfm_model_data_.bone[i].resize(temp_model->BoneData[i].BoneNum); // このメッシュのボーン数の領域確保
212 
213  for (unsigned j = 0; j < mfm_model_data_.bone[i].size(); j++)
214  {
215  // アニメーションフレーム数分領域確保
216  mfm_model_data_.bone[i][j].MatAry.resize(mfm_model_data_.nAnimFrame);
217 
218  Mtx init_matrix; // 初期姿勢行列
219  Mtx init_matrix_inverse; // ボーンオフセット行列
220 
221  // 初期姿勢行列の取得
222  const int MATRIX_WIDTH = 4;
223  const int MATRIX_HEIGHT = 4;
224  for (int r = 0; r < MATRIX_HEIGHT; r++)
225  {
226  for (int c = 0; c < MATRIX_WIDTH; c++)
227  {
228  init_matrix(r, c) = temp_model->BoneData[i].Data[j].init.Get(r, c);
229  }
230  }
231 
232  // ボーンオフセット行列(初期姿勢行列の逆行列)を作成
233  D3DXMatrixInverse(&init_matrix_inverse, NULL, &init_matrix);
234 
235  // 各アニメーションフレームのローカル姿勢変換行列の取得
236  for (int k = 0; k < mfm_model_data_.nAnimFrame; k++)
237  {
238  Mtx local_posture_conversion_matrix;
239 
240  for (int r = 0; r < MATRIX_HEIGHT; r++)
241  {
242  for (int c = 0; c < MATRIX_WIDTH; c++)
243  {
244  local_posture_conversion_matrix(r, c) = temp_model->BoneData[i].Data[j].perFrame.Data[k].Get(r, c);
245  }
246  }
247 
248  // ローカル変換行列にボーンオフセット行列を乗算
249  mfm_model_data_.bone[i][j].MatAry[k] = init_matrix_inverse * local_posture_conversion_matrix;
250  }
251  }
252  }
253 }

308 void ModelFBXObject::AccessVRAM(LPDIRECT3DDEVICE9 pDevice, std::vector<Shaderbuff>* temp_mesh)
309 {
310  // 頂点バッファの領域確保
311  mfm_model_data_.VertexBuffer.resize(temp_mesh->size());
312 
313  for (unsigned i = 0; i < temp_mesh->size(); i++)
314  {
315  // VRAMのメモリを確保(GPUに依頼)(頂点バッファの作成)
316  HRESULT hr = pDevice -> CreateVertexBuffer(sizeof(VERTEX_3D) * temp_mesh->at(i).Vertex.size(),
317  D3DUSAGE_WRITEONLY,
318  FVF_VERTEX_3D,
319  D3DPOOL_MANAGED,
320  &mfm_model_data_.VertexBuffer[i],
321  NULL);
322 
323  if (FAILED(hr))
324  {
325  MessageBox(NULL, "Vertex buffer was not created.", "Error", MB_OK);
326  }
327  }
328 
329  RegistrationVertex(temp_mesh);
330 
331  // 頂点宣言作成
332  pDevice->CreateVertexDeclaration(MFM_VERTEX_DECLARATION, &vertex_declaration_);
333 }

377 void ModelFBXObject::SetModelFBXObject()
378 {
379  // エフェクトファイルのオブジェクト取得
380  EffectFile* effect_file = EffectFile::GetPointer();
381 
382  effect_file->effect_->SetTechnique(shader_name_.c_str());
383 
384  // デバイスの取得
385  LPDIRECT3DDEVICE9 device = RendererDirectX9::GetDevice();
386  assert(device != nullptr && "Deviceの取得に失敗(ModelXObject.cpp)");
387 
388  // 頂点宣言の設定
389  device->SetVertexDeclaration(vertex_declaration_);
390 
391  Mtx** bone_matrix;
392  bone_matrix = new Mtx*[mfm_model_data_.VertexBuffer.size()];
393 
394  for (unsigned y = 0; y < mfm_model_data_.VertexBuffer.size(); y++)
395  {
396  bone_matrix[y] = new Mtx[mfm_model_data_.bone[y].size()];
397  for (unsigned x = 0; x < mfm_model_data_.bone[y].size(); x++)
398  {
399  bone_matrix[y][x] = mfm_model_data_.bone[y][x].MatAry[effect_file->animation_num_];
400  }
401  }
402 
403  MatrixSRT_Axis temp_inv_mtx;
404 
405  temp_inv_mtx.UpdateTransposeMatrix(&effect_file->world_matrix_);
406  temp_inv_mtx.TransposeMatrixTranslationOff();
407 
408  Mtx matrix_WVP = effect_file->world_matrix_ * effect_file->view_matrix_ * effect_file->projection_matrix_;
409 
410  // テクニックの開始
411  unsigned num_pass;
412 
413  HRESULT hr = effect_file->effect_->Begin(&num_pass, 0);
414 
415  for (unsigned pass = 0; pass < num_pass; pass++)
416  {
417  hr = effect_file->effect_->BeginPass(pass);
418 
419  if (FAILED(hr)) break;
420 
421  D3DLIGHT9 light;
422  // ライトカラー
423  light.Diffuse.r = 1.0f;
424  light.Diffuse.g = 1.0f;
425  light.Diffuse.b = 1.0f;
426  light.Diffuse.a = 1.0f;
427 
428  float brightess = 2.0f;
429 
430  D3DXVECTOR4 temp_vector;
431 
432  D3DXVec3Transform(&temp_vector, &light_vector_, &temp_inv_mtx.matrix_inverse_);
433 
434 
435  for (DWORD i = 0; i < mfm_model_data_.VertexBuffer.size(); i++)
436  {
437  effect_file->effect_->SetVector("directional_light_vector", (D3DXVECTOR4*)&temp_vector);
438  effect_file->effect_->SetVector("directional_light_diffuse",(D3DXVECTOR4*)&light.Diffuse);
439  effect_file->effect_->SetFloat ("directional_light_brightness", brightess);
440  effect_file->effect_->SetMatrix("matrix_world_View_projection", &matrix_WVP);
441  effect_file->effect_->SetMatrixArray("bones", bone_matrix[i], mfm_model_data_.bone[i].size());
442  effect_file->effect_->CommitChanges();
443  device -> SetStreamSource(0,
444  mfm_model_data_.VertexBuffer[i],
445  0,
446  sizeof(VERTEX_3D));
447 
448  // テクスチャセット
449  texture_manager_->SetTextureObject(&mfm_model_data_.MeshData[i].TexName, 0);
450 
451  // マテリアルをデバイスにセット
452  device->SetMaterial(&mfm_model_data_.Material[mfm_model_data_.MeshData[i].MaterialIndex]);
453 
454 
455  device->DrawPrimitive(D3DPT_TRIANGLELIST,
456  0,
457  mfm_model_data_.MeshData[i].PrimitiveNum);
458 
459 
460  }
461 
462  effect_file->effect_->EndPass();
463  }
464  effect_file->effect_->End();
465 
466  for (unsigned i = 0; i < mfm_model_data_.VertexBuffer.size(); i++)
467  {
468  delete[] bone_matrix[i];
469  }
470 
471  delete[] bone_matrix;
472 
473 }
