131 void Pause::Update()
132 {
133  Input();
134  icon_transform_.CreateWorldMatrix();
135  //icon_transform_.AddParentMatrixToWorldMatrix(getpTransform()->getpWorldMatrix());
136 
137  switch (select_type_)
138  {
139  case Pause::SelectType::CONTINUE:
140  {
141  character_color_[(int)Pause::SelectType::CONTINUE] = CHARACTER_COLOR1;
142  character_color_[(int)Pause::SelectType::RESTART] = CHARACTER_COLOR0;
143  character_color_[(int)Pause::SelectType::QUIT] = CHARACTER_COLOR0;
144  break;
145  }
146  case Pause::SelectType::RESTART:
147  {
148  character_color_[(int)Pause::SelectType::CONTINUE] = CHARACTER_COLOR0;
149  character_color_[(int)Pause::SelectType::RESTART] = CHARACTER_COLOR1;
150  character_color_[(int)Pause::SelectType::QUIT] = CHARACTER_COLOR0;
151  break;
152  }
153  case Pause::SelectType::QUIT:
154  {
155  character_color_[(int)Pause::SelectType::CONTINUE] = CHARACTER_COLOR0;
156  character_color_[(int)Pause::SelectType::RESTART] = CHARACTER_COLOR0;
157  character_color_[(int)Pause::SelectType::QUIT] = CHARACTER_COLOR1;
158  break;
159  }
160  }
161 
162  // 色の補間を進行
163  if (inter_container_[0].size() == 0) return;
164  XColor4 color = XColor4(*inter_iter_[0], *inter_iter_[1],
165  *inter_iter_[2], *inter_iter_[3]);
166  character_color_[(int)select_type_] = color;
167  inter_iter_[0]++;
168  inter_iter_[1]++;
169  inter_iter_[2]++;
170  inter_iter_[3]++;
171 }

175 void Pause::Input()
176 {
177  if (!getIsInput()) return;
178 
179  // 上
180  if (InputManager::getpInstance()->getpKeyboard()->getTrigger(DIK_W) ||
181  InputManager::getpInstance()->getpController()
182  ->getTriggerLStick(0, Controller::Direction::UP))
183  {
184  icon_transform_.getpPosition()->y += ADD_ICON_POSITION_Y;
185 
186  // ループ処理
187  if (icon_transform_.getpPosition()->y > TOP_ICON_POSITION_Y)
188  {
189  icon_transform_.getpPosition()->y = BOTTOM_ICON_POSITION_Y;
190  }
191 
192  SoundManager::getpInstance()->PlayOrStop(SoundManager::Type::SE_CLEAR);
193  }
194 
195  // 下
196  if (InputManager::getpInstance()->getpKeyboard()->getTrigger(DIK_S) ||
197  InputManager::getpInstance()->getpController()
198  ->getTriggerLStick(0, Controller::Direction::DOWN))
199  {
200  icon_transform_.getpPosition()->y += -ADD_ICON_POSITION_Y;
201 
202  // ループ処理
203  if (icon_transform_.getpPosition()->y < BOTTOM_ICON_POSITION_Y)
204  {
205  icon_transform_.getpPosition()->y = TOP_ICON_POSITION_Y;
206  }
207  SoundManager::getpInstance()->PlayOrStop(SoundManager::Type::SE_CLEAR);
208  }
209 
210  // 選択タイプ変更
211  if (icon_transform_.getpPosition()->y == TOP_ICON_POSITION_Y)
212  {
213  select_type_ = SelectType::CONTINUE;
214  }
215  else if (icon_transform_.getpPosition()->y == BOTTOM_ICON_POSITION_Y)
216  {
217  select_type_ = SelectType::QUIT;
218  }
219  else
220  {
221  select_type_ = SelectType::RESTART;
222  }
223 
224  // 決定
225  if (InputManager::getpInstance()->getpKeyboard()->getTrigger(DIK_SPACE) ||
226  InputManager::getpInstance()->getpController()
227  ->getTrigger(0, XINPUT_GAMEPAD_X))
228  {
229  // カラーアニメーションセット
230  float pattern_second = BLINKING_SECOND / BLINKING_NUM; // 1パターンの秒数
231  for (int i = 0; i < BLINKING_NUM; i++)
232  {
233  XColor4 color = (i % 2) == 0 ? CHARACTER_COLOR0 : CHARACTER_COLOR1;
234  double distance = (double)TimeToFrame::SecondToFrame(pattern_second);
235  inter_container_[0].push_back(color.r, distance);
236  inter_container_[1].push_back(color.g, distance);
237  inter_container_[2].push_back(color.b, distance);
238  inter_container_[3].push_back(color.a, distance);
239  }
240  inter_iter_[0] = inter_container_[0].begin();
241  inter_iter_[1] = inter_container_[1].begin();
242  inter_iter_[2] = inter_container_[2].begin();
243  inter_iter_[3] = inter_container_[3].begin();
244 
245  // 入力OFF
246  setIsInput(false);
247 
248  SoundManager::getpInstance()->PlayOrStop(SoundManager::Type::SE_KETEI);
249  }
250 }
