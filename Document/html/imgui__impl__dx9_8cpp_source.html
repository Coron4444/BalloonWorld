68  // Copy and convert all vertices into a single contiguous buffer

185 // Process Win32 mouse/keyboard inputs.
186 // You can read the io.WantCaptureMouse, io.WantCaptureKeyboard flags to tell if dear imgui wants to use your inputs.
187 // - When io.WantCaptureMouse is true, do not dispatch mouse input data to your main application.
188 // - When io.WantCaptureKeyboard is true, do not dispatch keyboard input data to your main application.
189 // Generally you may always pass all inputs to dear imgui, and hide them from your application based on those two flags.
190 // PS: In this Win32 handler, we use the capture API (GetCapture/SetCapture/ReleaseCapture) to be able to read mouse coordinations when dragging mouse outside of our window bounds.
191 IMGUI_API LRESULT ImGui_ImplWin32_WndProcHandler(HWND hwnd, UINT msg, WPARAM wParam, LPARAM lParam)
192 {
193  ImGuiIO& io = ImGui::GetIO();
194  switch (msg)
195  {
196  case WM_LBUTTONDOWN:
197  case WM_RBUTTONDOWN:
198  case WM_MBUTTONDOWN:
199  {
200  int button = 0;
201  if (msg == WM_LBUTTONDOWN) button = 0;
202  if (msg == WM_RBUTTONDOWN) button = 1;
203  if (msg == WM_MBUTTONDOWN) button = 2;
204  if (!IsAnyMouseButtonDown() && GetCapture() == NULL)
205  SetCapture(hwnd);
206  io.MouseDown[button] = true;
207  return 0;
208  }
209  case WM_LBUTTONUP:
210  case WM_RBUTTONUP:
211  case WM_MBUTTONUP:
212  {
213  int button = 0;
214  if (msg == WM_LBUTTONUP) button = 0;
215  if (msg == WM_RBUTTONUP) button = 1;
216  if (msg == WM_MBUTTONUP) button = 2;
217  io.MouseDown[button] = false;
218  if (!IsAnyMouseButtonDown() && GetCapture() == hwnd)
219  ReleaseCapture();
220  return 0;
221  }
222  case WM_MOUSEWHEEL:
223  io.MouseWheel += GET_WHEEL_DELTA_WPARAM(wParam) > 0 ? +1.0f : -1.0f;
224  return 0;
225  case WM_MOUSEMOVE:
226  io.MousePos.x = (signed short)(lParam);
227  io.MousePos.y = (signed short)(lParam >> 16);
228  return 0;
229  case WM_KEYDOWN:
230  case WM_SYSKEYDOWN:
231  if (wParam < 256)
232  io.KeysDown[wParam] = 1;
233  return 0;
234  case WM_KEYUP:
235  case WM_SYSKEYUP:
236  if (wParam < 256)
237  io.KeysDown[wParam] = 0;
238  return 0;
239  case WM_CHAR:
240  // You can also use ToAscii()+GetKeyboardState() to retrieve characters.
241  if (wParam > 0 && wParam < 0x10000)
242  io.AddInputCharacter((unsigned short)wParam);
243  return 0;
244  }
245  return 0;
246 }
