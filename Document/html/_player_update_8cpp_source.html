89 void PlayerUpdate::Update()
90 {
91  // 平行移動
92  Vector3D temp_vector;
93 
94  if (GetKeyboardPress(DIK_D))
95  {
96  temp_vector += *player_->GetTransform()->GetRight();
97  }
98  if (GetKeyboardPress(DIK_A))
99  {
100  temp_vector += -*player_->GetTransform()->GetRight();
101  }
102  if (GetKeyboardPress(DIK_W))
103  {
104  temp_vector += *player_->GetTransform()->GetForawrd();
105  }
106  if (GetKeyboardPress(DIK_S))
107  {
108  temp_vector += -*player_->GetTransform()->GetForawrd();
109  }
110 
111  *player_->GetPhysics()->GetAcceleration() += temp_vector.AnyLengthVector(SPEED);
112 
113  // ブレーキ
114  if (GetKeyboardTrigger(DIK_N))
115  {
116  //GameObjectManager::GetUpdateManager()
117  //->OverwriteArrayUpdateBase(player_, new PlayerUpdate_Special());
118  //return;
119  player_->GetPhysics()->AddFriction(0.5f);
120  }
121 
122  player_->GetPhysics()->AddFriction(0.9f);
123 
124  // ジャンプ
125  if (GetKeyboardTrigger(DIK_SPACE))
126  {
127  temp_vector = Vec3(0.0f, 1.0f, 0.0f);
128  temp_vector.AnyLengthVector(MeterPerSecond_To_MeterPerFlame(30.0f));
129 
130  *player_->GetPhysics()->GetVelocity() += temp_vector;
131  }
132 
133  // 拡縮
134  if (GetKeyboardPress(DIK_Z))
135  {
136  *player_->GetTransform()->GetScale() += Vec3(SPEED, SPEED, SPEED);
137  }
138  if (GetKeyboardPress(DIK_X))
139  {
140  *player_->GetTransform()->GetScale() += -Vec3(SPEED, SPEED, SPEED);
141  }
142 
143  player_->GetTransform()->SetLookAtSpeed(0.05f);
144 
145  // 回転
146  if (GetKeyboardPress(DIK_C))
147  {
148  player_->GetTransform()->SetLookAtVector(Vec3(1.0f, 1.0f, 0.0f));
149  }
150  if (GetKeyboardPress(DIK_V))
151  {
152  player_->GetTransform()->SetLookAtVector(Vec3(-1.0f, 0.0f, 0.0f));
153  }
154  if (GetKeyboardPress(DIK_B))
155  {
156  player_->GetTransform()->SetLookAtVector(Vec3(0.0f, 0.0f, -1.0f));
157  }
158 
159  float temp_scale = 1.0f;
160 
161  if (GetKeyboardTrigger(DIK_P))
162  {
163  *temp_object_->GetTransform()->GetPosition() = *player_->GetTransform()->GetPosition();
164  *temp_object_->GetTransform()->GetScale() = Vec3(temp_scale, temp_scale, temp_scale);
165  temp_scale += 1.0f;
166  temp_object_->GetTransform()->UpdateWorldMatrixSRT();
167  temp_object_->Play();
168  temp_object_->SetRepeat(true);
169 
170  //EffekseerManager::CreateDisposableEffekseerObject(&TEST_EFFEKSEER, *player_->GetTransform()->GetPosition());//Vec3(0.0f, 0.0f, 0.0f));
171  }
172 
173  player_->GetTransform()->UpdateAxisVector_WorldMatrixSRT();
174 }
