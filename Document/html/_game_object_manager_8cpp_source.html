159 void GameObjectManager::AddContentsOfAwaitAddArray()
160 {
161  // 追加待ちがあるかどうか
162  if (await_add_.GetEndPointer() <= 0) return;
163 
164  // 追加
165  for (unsigned i = 0; i < await_add_.GetEndPointer(); i++)
166  {
167  // 全体配列へ追加
168  all_game_object_.AddToArray(await_add_.GetArrayObject(i));
169 
170  // コンポーネントをマネージャにセット
171  SetComponentToManager(await_add_.GetArrayObject(i));
172  }
173 
174  // 追加待ち配列をリセット
175  await_add_.ResetArray();
176 }

181 // -解放待ち配列の中身を解放関数
182 //--------------------------------------------------
183 void GameObjectManager::ReleaseContentsOfAwaitReleaseArray()
184 {
185  // 解放待ちがあるかどうか
186  if (await_release_.GetEndPointer() <= 0) return;
187 
188  // 解放とソート
189  for (unsigned i = 0; i < await_release_.GetEndPointer(); i++)
190  {
191  // 全体配列からの消去
192  all_game_object_.DeleteFromArrayAndSortArray(await_release_.GetArrayObject(i));
193 
194  // オブジェクトの消去
195  GameObjectBase* temp = await_release_.GetArrayObject(i);
196  SafeRelease::Normal(&temp);
197  }
198 
199  // 解放待ち配列をリセット
200  await_release_.ResetArray();
201 }

266  }
267 
268  for (unsigned i = 0; i < all_game_object_.GetEndPointer(); i++)
269  {
270  ReleaseGameObjectBaseFromArray(all_game_object_.GetArrayObject(i));
271  }
272 
273  // 解放待ち配列の中身を解放
274  ReleaseContentsOfAwaitReleaseArray();
275 
276  // オブジェクト配列をリセット
277  all_game_object_.ResetArray();
