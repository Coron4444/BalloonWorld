149 inline void OverhangingPartOBB_Segment1D_Point(Vector3D* shortest_distance_vector,
150  Vector3D* obb_position,
151  Vector3D obb_vector_half,
152  Vector3D obb_axis_vector,
153  Vector3D* point)
154 {
155  // 長さが0以上か
156  float temp_length = obb_vector_half.GetLength();
157  if (temp_length <= 0.0f) return;
158 
159  // 点から直線に垂線を降ろしたときの交点の位置を表す値(tとする)を算出
160  Vector3D temp_vector = *point - *obb_position;
161  float t = temp_vector.CreateVectorDot(obb_axis_vector) / temp_length;
162 
163  // はみ出した部分のベクトルを算出
164  if (t < -1.0f)
165  {
166  *shortest_distance_vector += (t + 1.0f) * (obb_vector_half);
167  }
168  else if (t > 1.0f)
169  {
170  *shortest_distance_vector += (t - 1.0f) * (obb_vector_half);
171  }
172 }
