46 bool RendererDirectX9::Init(HINSTANCE hInstance, HWND hWnd, BOOL bWindow,
47  int nWindowWidth, int nWindowHeight)
48 {
49  // 使っていない引数の処理
50  hInstance = hInstance;
51 
52  // ダイレクト3Dインタフェースの取得
53  direct3d_interface_ = Direct3DCreate9(D3D_SDK_VERSION);
54 
55  if (direct3d_interface_ == nullptr) return false;
56 
57  // 現在のディスプレイモードを取得
58  D3DDISPLAYMODE direct3d_display_mode;
59  HRESULT h_result = direct3d_interface_->GetAdapterDisplayMode(D3DADAPTER_DEFAULT, // ディスプレイ・アダプタを指定する(今回の指定は初期のディスプレイ・アダプタ)
60  &direct3d_display_mode); // 現在のディスプレイ・アダプタの情報を受け取る変数の先頭アドレス
61 
62  if (FAILED(h_result)) return false;
63 
64  // デバイスのプレゼンテーションパラメータ(デバイスを作成するためのデータ)
65  D3DPRESENT_PARAMETERS direct3d_present_parameters;
66  ZeroMemory(&direct3d_present_parameters, sizeof(direct3d_present_parameters));
67  direct3d_present_parameters.BackBufferWidth = nWindowWidth; //　裏画面(バックバッファ)の幅
68  direct3d_present_parameters.BackBufferHeight = nWindowHeight; // 裏画面(バックバッファ)の高さ
69  direct3d_present_parameters.BackBufferFormat = direct3d_display_mode.Format; // 色数( バックバッファのフォーマット )
70  direct3d_present_parameters.BackBufferCount = 1; // バックバッファの数
71  direct3d_present_parameters.SwapEffect = D3DSWAPEFFECT_DISCARD; // 映像信号の同期(DISCARDは垂直同期を気にする)
72  direct3d_present_parameters.EnableAutoDepthStencil = TRUE; // TRUEで良い
73  direct3d_present_parameters.AutoDepthStencilFormat = D3DFMT_D24S8; // 今はこれで良い上のフォーマット
74  direct3d_present_parameters.Windowed = bWindow; // TRUEでウィンドウモード,FALSEでフルスクリーンモード
75  direct3d_present_parameters.FullScreen_RefreshRateInHz = D3DPRESENT_RATE_DEFAULT; // フルスクリーンモードの時のリフレッシュレートのHz数を変更
76  direct3d_present_parameters.PresentationInterval = D3DPRESENT_INTERVAL_DEFAULT; // モニターのHz数に合わせて描画される(IMMEDIATEだと垂直同期を待たずに描画する)( D3DPRESENT_INTERVAL_ONE　にするとヘルツで動く )
77 
78  // デバイスオブジェクトの生成( ｢描画｣と｢頂点処理｣をハードウェアで行う
79  h_result = direct3d_interface_->CreateDevice(D3DADAPTER_DEFAULT, // ディスプレイ・アダプタを指定する( 今回の指定は初期のディスプレイ・アダプタ )
80  D3DDEVTYPE_HAL, // デバイスタイプ( 今回の指定はハードウェアによる処理を行う )
81  hWnd, // このデバイスでフォーカスを設定するウィンドウハンドル
82  D3DCREATE_HARDWARE_VERTEXPROCESSING, // デバイスの作成を制御するフラグの組み合わせ( 今回の指定はハードウェアによる頂点処理を指定 )
83  &direct3d_present_parameters, // デバイスの設定を指定した変数の先頭アドレス
84  &direct3d_device_); // 作成したデバイスを受け取る変数の先頭アドレス
85 
86  if (FAILED(h_result))
87  {
88  MessageBox(hWnd, "ダイレクト3Dデバイスの\n作成に失敗しました!!", "Error", MB_OK);
89  return false;
90  }
91 
92  return true;
93 }

149 void RendererDirectX9::SetAlphaBlending_Linear()
150 {
151  // SRC --- 今から描くもの。
152  // DEST--- すでに描画されている画面のこと。
153  // 1は100%(255)のこと
154  // 今回はSRC_αが0だから背景色が100%で出ている。
155 
156  // {(1 - SRC_α) * DEST_RGB} + (SRC_RGB * SRC_α)
157  direct3d_device_->SetRenderState(D3DRS_BLENDOP, D3DBLENDOP_ADD);
158  direct3d_device_->SetRenderState(D3DRS_DESTBLEND, D3DBLEND_INVSRCALPHA);
159  direct3d_device_->SetRenderState(D3DRS_SRCBLEND, D3DBLEND_SRCALPHA);
160 }

232 void RendererDirectX9::SetDefaultSamplerState()
233 {
234  // 第二引数(UVの設定)
235  // 第三引数テクスチャー座標外の処理
236  // WRAP ( 複製 )
237  // CLAMP( 一番最後のピクセルを引き延ばして参照する )
238  // MIRROR( 鏡 )
239  direct3d_device_->SetSamplerState(0, D3DSAMP_ADDRESSU, D3DTADDRESS_WRAP);
240  direct3d_device_->SetSamplerState(0, D3DSAMP_ADDRESSV, D3DTADDRESS_WRAP);
241 
242  // フィルタリング
243  // MIPマップ(あらかじめ小さい画像データを自動で用意しておく)
244  direct3d_device_->SetSamplerState(0, D3DSAMP_MINFILTER, D3DTEXF_LINEAR); // デフォルトはPOINT(代表点1個を拾ってくる)
245  // LINEAR(周りの点を集約して、つまり色をブレンドして1点にする)よって色がぼけるが滑らか
246  direct3d_device_->SetSamplerState(0, D3DSAMP_MAGFILTER, D3DTEXF_LINEAR); // デフォルトはPOINT
247  direct3d_device_->SetSamplerState(0, D3DSAMP_MIPFILTER, D3DTEXF_LINEAR); // デフォルトはNONE(MIPマップを使わない)
248  // LINEAR（MIPマップ間を線形補完する）
249 }
