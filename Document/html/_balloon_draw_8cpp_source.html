83 void BalloonDraw::Init()

92  //getpDrawOrderList()->setVertexShaderType(ShaderManager::VertexShaderType::DEFAULT_SHADOW);
93  //getpDrawOrderList()->setPixelShaderType(ShaderManager::PixelShaderType::DEFAULT_SHADOW);
94  getpDrawOrderList()->setVertexShaderType(ShaderManager::VertexShaderType::BLINN_PHONG);
95  getpDrawOrderList()->setPixelShaderType(ShaderManager::PixelShaderType::BLINN_PHONG);
96 
97 
98  // オブジェクト取得
99  balloon_object_ = ModelXManager::getpInstance()->getpObject(&BALLOON_MODEL_NAME);
100  balloon_line_object_ = ModelXManager::getpInstance()->getpObject(&BALLOON_LINE_MODEL_NAME);
101 
102  // オブジェクト変形初期化
103  object_transform_.resize(balloon_->getAllObjectNum());
104  for (int i = 0; i < balloon_->getAllObjectNum() - 1; i++)
105  {
106  *object_transform_[i].getpScale() = Balloon::OBB_EDGE_LENGTH_HALF * 2.5f;
107  }
108  float adjustment_balloon_scale = 0.9f;
109  object_transform_[balloon_->getAllObjectNum() - 1].getpScale()->x
110  = Balloon::SPHERE_RADIUS * adjustment_balloon_scale;
111  object_transform_[balloon_->getAllObjectNum() - 1].getpScale()->y
112  = Balloon::SPHERE_RADIUS;
113  object_transform_[balloon_->getAllObjectNum() - 1].getpScale()->z
114  = Balloon::SPHERE_RADIUS * adjustment_balloon_scale;
115 
116  // 色初期化
117  int color_index = Random::getpInstance()->getInt(0, BALLOON_COLOR_NUM - 1);
118  for (unsigned i = 0; i < balloon_object_->getMeshNum(); i++)
119  {
120  color_ = BALLOON_COLOR[color_index];
121  }
122 }
