122 void Capsule::CalculationMinAndMax()
123 {
124  Vector3D max;
125  Vector3D min;
126 
127  // X座標
128  Axis axis;
129  axis.RotationAxisZ(D3DXToRadian(-45.0f));
130  Vector3D up = *axis.getpUp();
131  Vector3D right = Vector3D(1.0f, 0.0f, 0.0f);
132  up.ChangeAnyLength(getpVector()->getLength());
133  right.ChangeAnyLength(getRadius());
134 
135  Vector3D temp_max = *getpPosition() + up + right;
136  max.x = temp_max.x;
137  Vector3D temp_min = *getpPosition() + -right;
138  min.x = temp_min.x;
139 
140  // Y座標
141  right = Vector3D(0.0f, 1.0f, 0.0f);
142  right.ChangeAnyLength(getRadius());
143  temp_max = *getpPosition() + up + right;
144  max.y = temp_max.y;
145  temp_min = *getpPosition() + -right;
146  min.y = temp_min.y;
147 
148  // Z座標
149  axis.Reset();
150  axis.RotationAxisX(D3DXToRadian(45.0f));
151  up = *axis.getpUp();
152  Vector3D forward = Vector3D(0.0f, 0.0f, 1.0f);
153  up.ChangeAnyLength(getpVector()->getLength());
154  forward.ChangeAnyLength(getRadius());
155  temp_max = *getpPosition() + up + forward;
156  max.z = temp_max.z;
157  temp_min = *getpPosition() + -forward;
158  min.z = temp_min.z;
159 
160  *getpMax() = max;
