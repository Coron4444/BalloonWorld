190 void BulletPhysicsManager::Init()
191 {
192  // 衝突計算の設定(今回はデフォルトを使用)
193  configuration_ = new btDefaultCollisionConfiguration();
194  dispatcher_ = new btCollisionDispatcher(configuration_);
195 
196  // ブロードフェーズ法(衝突する可能性のあるオブジェクトの検出)の設定(今回はOctree)
197  broadphase_ = new btDbvtBroadphase();
198 
199  // 拘束ソルバ設定(オブジェクトの重なり解消などの設定)
200  solver_ = new btSequentialImpulseConstraintSolver();
201 
202  // bulletワールド設定
203  dynamics_world_ = new btDiscreteDynamicsWorld(dispatcher_, broadphase_, solver_, configuration_);
204  dynamics_world_->setGravity(DEFULT_GRAVITY);
205 
206  // デバッグ初期化
207  debug_ = new BulletPhysicsDebug();
208  debug_->Init();
209  dynamics_world_->setDebugDrawer(debug_);
210 }

214 void BulletPhysicsManager::Uninit()
215 {
216  // 拘束の解放
217  auto iterator2 = constraint_.begin();
218  while (iterator2 != constraint_.end())
219  {
220  (*iterator2)->ForcedRelease();
221  iterator2 = constraint_.begin();
222  }
223  constraint_.clear();
224 
225  // ワールドの解放
226  dynamics_world_->setDebugDrawer(nullptr);
227  SafeRelease::Normal(&dynamics_world_);
228 
229  // オブジェクトの解放
230  auto iterator = object_.begin();
231  while (iterator != object_.end())
232  {
233  (*iterator)->ForcedRelease();
234  iterator = object_.begin();
235  }
236  object_.clear();
237 
238  // その他解放
239  SafeRelease::Normal(&debug_);
240  SafeRelease::Normal(&solver_);
241  SafeRelease::Normal(&broadphase_);
242  SafeRelease::Normal(&dispatcher_);
243  SafeRelease::Normal(&configuration_);
244 }
