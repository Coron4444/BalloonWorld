108 inline float ShortestDistancePointAndLine(Vec3* cross_point, float* t, Vector3D* point,
109  LineSegment* line_segment)
110 {
111  // 点と線分の始点のベクトル
112  Vector3D temp_vector = *point - *line_segment->getpPosition();
113 
114  // 線分が0以上かどうか
115  if (line_segment->getpVector()->GetLengthSquare() > 0.0f)
116  {
117  // 交点のtの算出
118  *t = line_segment->getpVector()->CreateVectorDot(temp_vector)
119  / line_segment->getpVector()->CreateVectorDot(*line_segment->getpVector());
120  }
121  else
122  {
123  *t = 0.0f;
124  }
125 
126  // 交点のhの算出
127  *cross_point = *line_segment->getpPosition() + *line_segment->getpVector() * (*t);
128 
129  // 距離を返す
130  temp_vector = *cross_point - *point;
131  return temp_vector.GetLength();
132 }

208 inline float ShortestDistance_Line_Line(Vec3* cross_point0, Vec3* cross_point1,
209  float* t0, float* t1,
210  LineSegment* line_segment0,
211  LineSegment* line_segment1)
212 {
213  // 2直線が平行
214  if (Is_ParallelSegment(line_segment0, line_segment1))
215  {
216  // 直線1と直線0の座標との最短距離を算出
217  float length = ShortestDistancePointAndLine(cross_point1, t1,
218  line_segment0->getpPosition(),
219  line_segment1);
220  *cross_point0 = *line_segment0->getpPosition();
221  *t0 = 0.0f;
222 
223  return length;
224  }
225 
226  // 平行でない
227  Vector3D between_position_vector = *line_segment0->getpPosition()
228  - *line_segment1->getpPosition();
229 
230  float dot_vector0_vector1 = line_segment0->getpVector()
231  ->CreateVectorDot(*line_segment1->getpVector());
232  float dot_vector0_vector0 = line_segment0->getpVector()->GetLengthSquare();
233  float dot_vector1_vector1 = line_segment1->getpVector()->GetLengthSquare();
234  float dot_vector0_position = line_segment0->getpVector()
235  ->CreateVectorDot(between_position_vector);
236  float dot_vector1_position = line_segment1->getpVector()
237  ->CreateVectorDot(between_position_vector);
238 
239  *t0 = (dot_vector0_vector1 * dot_vector1_position - dot_vector1_vector1 * dot_vector0_position) / (dot_vector0_vector0 * dot_vector1_vector1 - dot_vector0_vector1 * dot_vector0_vector1);
240  *cross_point0 = *line_segment0->getpPosition() + *line_segment0->getpVector() * (*t0);
241 
242  Vector3D temp_vector = *cross_point0 - *line_segment1->getpPosition();
243 
244  *t1 = line_segment1->getpVector()->CreateVectorDot(temp_vector) / dot_vector1_vector1;
245 
246  *cross_point1 = *line_segment1->getpPosition() + *line_segment1->getpVector() * (*t1);
247 
248  temp_vector = *cross_point1 - *cross_point0;
249 
250  return temp_vector.GetLength();
251 }
