256  XAUDIO2_BUFFER buffer;
257 
258  // バッファの値設定
259  memset(&buffer, 0, sizeof(XAUDIO2_BUFFER));
260  buffer.AudioBytes = size_audio_[(int)type];
261  buffer.pAudioData = data_audio_[(int)type];
262  buffer.Flags = XAUDIO2_END_OF_STREAM;

264 
265  // 状態取得
266  source_voice_[(int)type]->GetState(&xa2state);
267  if (xa2state.BuffersQueued != 0)
268  {// 再生中
269  // 一時停止
270  source_voice_[(int)type]->Stop(0);
271 
272  // オーディオバッファの削除
273  source_voice_[(int)type]->FlushSourceBuffers();
274  }
275 
276  // オーディオバッファの登録
277  source_voice_[(int)type]->SubmitSourceBuffer(&buffer);
278 
279  // 再生
280  source_voice_[(int)type]->Start(0);
281 
282  return S_OK;
283 }

320 HRESULT SoundManager::CheckChunk(HANDLE file_handle, DWORD format, DWORD* chunk_size,
321  DWORD* chunk_data_position)
322 {
323  HRESULT hr = S_OK;
324  DWORD dwRead;
325  DWORD dwChunkType;
326  DWORD dwChunkDataSize;
327  DWORD dwRIFFDataSize = 0;
328  DWORD dwFileType;
329  DWORD dwBytesRead = 0;
330  DWORD dwOffset = 0;
331 
332  if (SetFilePointer(file_handle, 0, NULL, FILE_BEGIN) == INVALID_SET_FILE_POINTER)
333  {// ファイルポインタを先頭に移動
334  return HRESULT_FROM_WIN32(GetLastError());
335  }
336 
337  while (hr == S_OK)
338  {
339  if (ReadFile(file_handle, &dwChunkType, sizeof(DWORD), &dwRead, NULL) == 0)
340  {// チャンクの読み込み
341  hr = HRESULT_FROM_WIN32(GetLastError());
342  }
343 
344  if (ReadFile(file_handle, &dwChunkDataSize, sizeof(DWORD), &dwRead, NULL) == 0)
345  {// チャンクデータの読み込み
346  hr = HRESULT_FROM_WIN32(GetLastError());
347  }
348 
349  switch (dwChunkType)
350  {
351  case 'FFIR':
352  dwRIFFDataSize = dwChunkDataSize;
353  dwChunkDataSize = 4;
354  if (ReadFile(file_handle, &dwFileType, sizeof(DWORD), &dwRead, NULL) == 0)
355  {// ファイルタイプの読み込み
356  hr = HRESULT_FROM_WIN32(GetLastError());
357  }
358  break;
359 
360  default:
361  if (SetFilePointer(file_handle, dwChunkDataSize, NULL, FILE_CURRENT) == INVALID_SET_FILE_POINTER)
362  {// ファイルポインタをチャンクデータ分移動
363  return HRESULT_FROM_WIN32(GetLastError());
364  }
365  }
366 
367  dwOffset += sizeof(DWORD) * 2;
368  if (dwChunkType == format)
369  {
370  *chunk_size = dwChunkDataSize;
371  *chunk_data_position = dwOffset;
372 
373  return S_OK;
374  }
375 
376  dwOffset += dwChunkDataSize;
377  if (dwBytesRead >= dwRIFFDataSize)
378  {
379  return S_FALSE;
380  }
381  }
382 
383  return S_OK;
384 }
