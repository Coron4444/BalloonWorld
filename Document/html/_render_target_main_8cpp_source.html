169 void RenderTargetMain::SortTransparent()
170 {
171  // 透明オブジェクトがあるかどうか
172  if (draw_transparent_.getEndIndex() <= 0) return;
173 
174  for (unsigned i = 0; i < draw_transparent_.getEndIndex() - 1; i++)
175  {
176  // 深度値を算出
177  Vector3D temp_vector0 = *draw_transparent_.getObject(i)->getpGameObject()
178  ->getpTransform()->getpPosition() - *camera_->getpPosition();
179 
180  float depth_value0 = temp_vector0.getLengthSquare();
181 
182  for (unsigned j = i + 1; j < draw_transparent_.getEndIndex(); j++)
183  {
184  // 深度値を算出
185  Vector3D temp_vector1 = *draw_transparent_.getObject(j)->getpGameObject()
186  ->getpTransform()->getpPosition() - *camera_->getpPosition();
187 
188  float depth_value1 = temp_vector1.getLengthSquare();
189 
190  // 深度値を比較
191  if (depth_value0 < depth_value1)
192  {
193  // 並び替え
194  draw_transparent_.SortTheTwoObject(i, j);
195  }
196  }
197  }
198 }
