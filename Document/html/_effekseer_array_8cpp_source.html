146 void EffekseerArray::CreateDisposableEffekseerObject(Vec3 position, Vec3 scale)
147 {
148  // 停止しているオブジェクトを探す
149  for (auto contents : disposable_effekseer_array_)
150  {
151  if (contents->IsPlaing()) continue;
152 
153  // リストへ追加
154  disposable_effekseer_list_.push_back(contents);
155 
156  // 状態更新
157  *contents->GetTransform()->GetPosition() = position;
158  *contents->GetTransform()->GetScale() = scale;
159  contents->GetTransform()->UpdateWorldMatrixSRT();
160 
161  // 再生
162  contents->Play();
163 
164  break;
165  }
166 }

191 void EffekseerArray::UpdateList()
