41  ShaderBase::GetDevice()->SetTextureStageState(0, D3DTSS_ALPHAARG2, D3DTA_DIFFUSE);
42

126 void VertexShaderFixed::UpdateDirectionalLignt()
127 {
128  // ライトの設定
129  ZeroMemory(&directional_light_, sizeof(directional_light_));
130 
131  directional_light_.Type = D3DLIGHT_DIRECTIONAL;
132 
133  // ライト方向を単位ベクトルで代入
134  Vec3 directional_light_vector = *GameObjectManager::GetDrawManager()->GetDirectionalLightVector();
135  D3DXVec3Normalize((D3DXVECTOR3*)&directional_light_.Direction, &directional_light_vector);
136 
137  // ライトカラー
138  directional_light_.Diffuse.r = 1.0f;
139  directional_light_.Diffuse.g = 1.0f;
140  directional_light_.Diffuse.b = 1.0f;
141  directional_light_.Diffuse.a = 1.0f;
142 
143  // アンビエントライト
144  directional_light_.Ambient.r = 0.1f;
145  directional_light_.Ambient.g = 0.1f;
146  directional_light_.Ambient.b = 0.1f;
147  directional_light_.Ambient.a = 1.0f;
148 
149  // グローバルアンビエントの設定
150  ShaderBase::GetDevice()->SetRenderState(D3DRS_AMBIENT, D3DXCOLOR(0.1f, 0.1f, 0.1f, 1.0f));
151 
152  // ライトをデバイスにセット(第一引数は0〜4のライト番号)
153  ShaderBase::GetDevice()->SetLight(0, &directional_light_);
154 
155  // 0番ライトを使えるようにする
156  ShaderBase::GetDevice()->LightEnable(0, TRUE);
157 }
