265 void SceneManager::SceneChange()
266 {
267  // フェード処理が終わっているかどうか
268  if (!GameObjectManager::getpInstance()->getpDrawManager()->IsFadeEnd()) return;
269 
270  // フェードのステートごとの処理
271  if (GameObjectManager::getpInstance()->getpDrawManager()
272  ->IsFadeState(Fade::State::FADE_OUT))
273  {
274  // 現在のメインシーンの終了処理
275  SafeRelease::PlusUninit(&current_scene_);
276 
277  // ゲームオブジェクトマネージャのシーン変更時終了処理
278  GameObjectManager::getpInstance()->UninitWhenChangeScene();
279 
280  // 新しいシーンへ移行
281  current_scene_ = next_scene_;
282  next_scene_ = nullptr;
283  current_scene_->setSceneManager(this);
284  current_scene_->Init();
285 
286  // フェードイン初期化
287  GameObjectManager::getpInstance()->getpDrawManager()
288  ->StartFadeIn(fade_type_,
289  Vec2((float)GameEngine::SCREEN_WIDTH, (float)GameEngine::SCREEN_HEIGHT),
290  fade_color_, fade_speed_);
291  }
292  else if (GameObjectManager::getpInstance()->getpDrawManager()
293  ->IsFadeState(Fade::State::FADE_IN))
294  {
295  // フェード終了処理
296  state_ = SceneManager::State::NONE;
297 
298  // フェード中フラグOFF
299  is_fade_ = false;
300  }
301 }
