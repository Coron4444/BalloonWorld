147 void DrawManager::OverwriteArrayDrawBase(GameObjectBase* game_object,
148  DrawBase* new_draw)
149 {
150  // 新規がnullptrの場合
151  if (new_draw == nullptr)
152  {
153  // 古い描画基底クラスの解放
154  ReleaseDrawBaseFromArray(game_object->GetDraw());
155 
156  // 古い描画基底クラスの消去
157  DrawBase* temp = game_object->GetDraw();
158  SafeRelease::Normal(&temp);
159 
160  // nullptrの代入
161  game_object->SetDraw(new_draw);
162  }
163  else
164  {
165  // 配列の上書き
166  all_draw_.OverwriteArray(game_object->GetDraw(), new_draw);
167 
168  // 古い描画基底クラスの消去
169  DrawBase* temp = game_object->GetDraw();
170  SafeRelease::Normal(&temp);
171 
172  // 新規コンポーネントの初期化
173  game_object->SetDraw(new_draw);
174  *new_draw->getpGameObject() = *game_object;
175  new_draw->Init();
176  }
177 }
