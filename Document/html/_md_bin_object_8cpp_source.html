340  material_[i].Diffuse.g = *md_bin_data->getpMaterial(i)->getpDiffuse()->getpG();

342  material_[i].Diffuse.a = alpha;

344  // エミッシブ
345  material_[i].Emissive.r = *md_bin_data->getpMaterial(i)->getpEmissive()->getpR();
346  material_[i].Emissive.g = *md_bin_data->getpMaterial(i)->getpEmissive()->getpG();
347  material_[i].Emissive.b = *md_bin_data->getpMaterial(i)->getpEmissive()->getpB();
348  material_[i].Emissive.a = alpha;
349 
350  // スペキュラ&スペキュラパワー
351  material_[i].Specular.r = *md_bin_data->getpMaterial(i)->getpSpecular()->getpR();
352  material_[i].Specular.g = *md_bin_data->getpMaterial(i)->getpSpecular()->getpG();
353  material_[i].Specular.b = *md_bin_data->getpMaterial(i)->getpSpecular()->getpB();

540 void MdBinObject::CreateBone(int mesh_index,
541  MdBinData* md_bin_data)
542 {
543  int bone_num = md_bin_data->getpMesh(mesh_index)->getBoneArraySize();
544  if (bone_num > 0)
545  {
546  mesh_[mesh_index].setBoneArraySize(bone_num);
547  for (int i = 0; i < bone_num; i++)
548  {
549  // ボーン名
550  *mesh_[mesh_index].getpBone(i)->getpName() = *md_bin_data->getpMesh(mesh_index)->getpBone(i)->getpName();
551 
552  // オフセット行列
553  ChangeMatrix(mesh_[mesh_index].getpBone(i)->getpOffsetMatrix(),
554  md_bin_data->getpMesh(mesh_index)->getpBone(i)->getpOffsetMatrix());
555 
556  Matrix offset_matrix;
557  D3DXMatrixInverse(&offset_matrix, NULL, mesh_[mesh_index].getpBone(i)->getpOffsetMatrix());
558 
559  // アニメーション行列
560  int animation_fram_num = md_bin_data->getAnimationFramNum();
561  mesh_[mesh_index].getpBone(i)->setAnimationMatrixArraySize(animation_fram_num);
562  for (int j = 0; j < animation_fram_num; j++)
563  {
564  Matrix frame_matrix;
565  ChangeMatrix(&frame_matrix,
566  md_bin_data->getpMesh(mesh_index)->getpBone(i)->getpAnimationMatrix(j));
567 
568  // オフセット行列とフレーム行列を乗算しアニメーション行列とする
569  *mesh_[mesh_index].getpBone(i)->getpAnimationMatrix(j) = offset_matrix * frame_matrix;
570  }
571  }
572  }
573  else
574  {
575  // ボーンがない場合はデフォルトボーンを1本持たせる
576  bone_num = 1;
577  mesh_[mesh_index].setBoneArraySize(bone_num);
578 
579  // ボーン名
580  *mesh_[mesh_index].getpBone(0)->getpName() = "DefaultBone";
581 
582  // オフセット行列
583  Matrix default_matrix;
584  D3DXMatrixIdentity(&default_matrix);
585  *mesh_[mesh_index].getpBone(0)->getpOffsetMatrix() = default_matrix;
586 
587  // アニメーション行列
588  int animation_fram_num = md_bin_data->getAnimationFramNum();
589  mesh_[mesh_index].getpBone(0)->setAnimationMatrixArraySize(animation_fram_num);
590  for (int j = 0; j < animation_fram_num; j++)
591  {
592  *mesh_[mesh_index].getpBone(0)->getpAnimationMatrix(j) = default_matrix;
593  }
594  }

646 void MdBinObject::CreateVertexBuffer(int mesh_index)
647 {
648  // VRAMのメモリを確保(GPUに依頼)(頂点バッファの作成)
649  unsigned buffer_num = sizeof(MdBinObject::Vertex) * mesh_[mesh_index].getVertexArraySize();
650  HRESULT hr = device_->CreateVertexBuffer(buffer_num, // 借りたいbafの量(バイト)、つまり1頂点の容量×必要頂点数
651  D3DUSAGE_WRITEONLY, // 使用用途(今回は書き込みのみ、GPUが早く動くが書き込んだデータを読んではダメ(値が不定))
652  RendererDirectX9::FVF_VERTEX_3D, // 頂点属性
653  D3DPOOL_MANAGED, // 頂点バッファの管理方法( MANAGEDは管理はDirect3Dにお任せという意味 )
654  mesh_[mesh_index].getp2VertexBuffer(), // 管理者の居場所のメモ帳(ポインタのポインタ)(全てはこれの値を知るための作業)
655  NULL); // なぞ？
656  if (FAILED(hr))
657  {
658  MessageBox(NULL, "頂点バッファの生成に失敗(MdBinObject.cpp)", "Error", MB_OK);
659  }
660 
661  // 頂点の登録
662  MdBinObject::Vertex* temp_vertex; // 仮想アドレスをもらう頂点ポインタ
663 
664  // ロックする
665  mesh_[mesh_index].getpVertexBuffer()->Lock(0, // どこからロックしたいか
666  0, // どのくらい借りたいか(0はVertexBuffer全部)
667  (void**)&temp_vertex, // ロックするとVRAMの仮想アドレスがもらえる(実際に書き込んでいるわけではない)
668  D3DLOCK_DISCARD); // 0でもよい
669 
670  for (auto i = 0; i < mesh_[mesh_index].getVertexArraySize(); i++)
671  {
672  temp_vertex[i] = *mesh_[mesh_index].getpVertex(i);
673  }
674 
675  // アンロック
676  mesh_[mesh_index].getpVertexBuffer()->Unlock();
677 }

681 void MdBinObject::CreateIndexBuffer(int mesh_index)
682 {
683  // VRAMのメモリを確保(GPUに依頼)(インデックスバッファの作成)
684  unsigned buffer_num = sizeof(WORD) * mesh_[mesh_index].getIndexArraySize();
685  HRESULT hr = device_->CreateIndexBuffer(buffer_num, // 借りたいbafの量(バイト)、つまり1頂点の容量×必要頂点数
