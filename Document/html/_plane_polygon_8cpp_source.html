129 void PlanePolygon::Draw()
130 {
131  // FVFの設定(デバイスの方に頂点の属性を教える)
132  device_->SetFVF(RendererDirectX9::FVF_VERTEX_3D);
133 
134  // 頂点バッファを使うGPUとバーテックスバッファのパイプライン
135  // 使いたくなかったらNULLをいれる
136  device_->SetStreamSource(0, // パイプライン番号
137  vertex_buffer_, // バーテックスバッファ変数名
138  0, // どこから流し込むか
139  sizeof(RendererDirectX9::VERTEX_3D)); // ストライド値(隣の頂点までの長さ＝1頂点の大きさ)
140 
141  // インデックスをセットする
142  device_->SetIndices(index_buffer_);
143 
144  // ライティングをONにする(ライト全てをON、OFFする)
145  device_->SetRenderState(D3DRS_LIGHTING, FALSE);
146 
147  device_->DrawIndexedPrimitive(D3DPT_TRIANGLESTRIP,
148  0, // セットストリームからどれくらいずれているか
149  0, // インデックスで一番小さい値
150  (UINT)vertex_.size(), // 頂点数
151  0, // スタートインデックス番号
152  PRIMITIVE_NUM); // プリミティブ数
153 
154 
155  // ライティングをONにする(ライト全てをON、OFFする)
156  device_->SetRenderState(D3DRS_LIGHTING, TRUE);
157 }

167 void PlanePolygon::CreateVertex(Color4 color, Vec2 plane_size)
168 {
169  // 頂点情報の入力
170  float temp_x = plane_size.x / 2;
171  float temp_y = plane_size.y / 2;
172 
173  float u0 = 0.0f;
174  float v0 = 0.0f;
175  float u1 = 1.0f;
176  float v1 = 1.0f;
177 
178  vertex_.push_back({Vec3(-temp_x, temp_y, 0.0f), Vec3(0.0f, 0.0f, -1.0f), color, Vec2(u0, v0)});
179  vertex_.push_back({Vec3(temp_x, temp_y, 0.0f), Vec3(0.0f, 0.0f, -1.0f), color, Vec2(u1, v0)});
180  vertex_.push_back({Vec3(-temp_x, -temp_y, 0.0f), Vec3(0.0f, 0.0f, -1.0f), color, Vec2(u0, v1)});
181  vertex_.push_back({Vec3(temp_x, -temp_y, 0.0f), Vec3(0.0f, 0.0f, -1.0f), color, Vec2(u1, v1)});
182 }

234 void PlanePolygon::AccessVRAM()
235 {
236  // VRAMのメモリを確保(GPUに依頼)(頂点バッファの作成)
237  HRESULT hr = device_->CreateVertexBuffer(sizeof(RendererDirectX9::VERTEX_3D) * (int)vertex_.size(), // 借りたいbafの量(バイト)、つまり1頂点の容量×必要頂点数
238  D3DUSAGE_WRITEONLY, // 使用用途(今回は書き込みのみ、GPUが早く動くが書き込んだデータを読んではダメ(値が不定))
239  RendererDirectX9::FVF_VERTEX_3D, // 頂点属性
240  D3DPOOL_MANAGED, // 頂点バッファの管理方法( MANAGEDは管理はDirect3Dにお任せという意味 )
241  &vertex_buffer_, // 管理者の居場所のメモ帳(ポインタのポインタ)(全てはこれの値を知るための作業)
242  NULL);
243 
244  if (FAILED(hr))
245  {
246  MessageBox(NULL, "Vertex buffer was not created.", "Error", MB_OK);
247  }
248 
249 
250  // VRAMのメモリを確保(GPUに依頼)(インデックスバッファの作成)
251  hr = device_->CreateIndexBuffer(sizeof(WORD) * (int)index_.size(), // 借りたいbafの量(バイト)、つまり1頂点の容量×必要頂点数
252  D3DUSAGE_WRITEONLY, // 使用用途(今回は書き込みのみ、GPUが早く動くが書き込んだデータを読んではダメ(値が不定))
253  D3DFMT_INDEX16, // 頂点フォーマット(WORD型だから16、DWORD型なら32)
254  D3DPOOL_MANAGED, // 頂点バッファの管理方法( MANAGEDは管理はDirect3Dにお任せという意味 )
255  &index_buffer_, // 管理者の居場所のメモ帳(ポインタのポインタ)(全てはこれの値を知るための作業)
256  NULL); // なぞ？
257 
258  if (FAILED(hr))
259  {
260  MessageBox(NULL, "Index buffer was not created.", "Error", MB_OK);
261  }
262 
263  RegistrationVertex();
264 
265  RegistrationIndex();
266 }
