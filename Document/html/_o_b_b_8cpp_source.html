263 void OBB::CalculationMinAndMax()
264 {
265  Vector3D point[MAX_POINT_NUM];
266  point[0] = position_ + getRightLengthVectorHalf() + getUpLengthVectorHalf() + getForwardLengthVectorHalf();
267  point[1] = position_ + getRightLengthVectorHalf() + getUpLengthVectorHalf() + -getForwardLengthVectorHalf();
268  point[2] = position_ + -getRightLengthVectorHalf() + getUpLengthVectorHalf() + getForwardLengthVectorHalf();
269  point[3] = position_ + -getRightLengthVectorHalf() + getUpLengthVectorHalf() + -getForwardLengthVectorHalf();
270  point[4] = position_ + getRightLengthVectorHalf() + -getUpLengthVectorHalf() + getForwardLengthVectorHalf();
271  point[5] = position_ + getRightLengthVectorHalf() + -getUpLengthVectorHalf() + -getForwardLengthVectorHalf();
272  point[6] = position_ + -getRightLengthVectorHalf() + -getUpLengthVectorHalf() + getForwardLengthVectorHalf();
273  point[7] = position_ + -getRightLengthVectorHalf() + -getUpLengthVectorHalf() + -getForwardLengthVectorHalf();
274 
275  Vector3D max = point[0];
276  Vector3D min = point[0];
277  for (int i = 0; i < MAX_POINT_NUM; i++)
278  {
279  if (max.x < point[i].x) max.x = point[i].x;
280  if (max.y < point[i].y) max.y = point[i].y;
281  if (max.z < point[i].z) max.z = point[i].z;
282  if (min.x > point[i].x) min.x = point[i].x;
283  if (min.y > point[i].y) min.y = point[i].y;
284  if (min.z > point[i].z) min.z = point[i].z;
285  }
286 
287  *getpMax() = max;
288  *getpMin() = min;
289 }
